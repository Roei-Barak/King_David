# פֶּרֶק יח: לֹא אַתָּה תִּבְנֶה

הארמון בירושלים שקט היה באותו לילה. שנים אחדות חלפו מאז עלה דוד אל העיר, ושלום יחסי שרר בגבולות. דוד ישב והביט בחלון, ובלבו רחשה אי־נחת של מי שחש עוול שאינו עוול לו. הוא קרא לנתן.

כשבא הנביא, דיבר אליו דוד בהתלהבות אמיתית, לא כהצגה של מלך אלא כאיש שדבר אחד מטריד את מנוחתו: "רְאֵה נָא — אָנֹכִי יוֹשֵׁב בְּבֵית אֲרָזִים (שמ"ב ז, ב), וַאֲרוֹן הָאֱלֹהִים יֹשֵׁב בְּתוֹךְ הַיְרִיעָה!" הוא הביט בנתן, ואי הנחת ניכרה בו. "לִי אֲרָזִים — וְלוֹ יְרִיעוֹת. לִי קִירוֹת — וְלוֹ רוּחַ." שתק רגע, ואז: "הֲנָאֶה הַדָּבָר בְּעֵינֶיךָ — כִּי אֵשֵׁב בַּמָּלֵא וְאָרוֹן ה' בַּדָּל?"

נתן חייך בקלות, שכן עוד לא בא אליו דבר ה'. "כֹּל אֲשֶׁר בִּלְבָבְךָ — לֵךְ עֲשֵׂה," אמר, "כִּי ה' עִמָּךְ" (שמ"ב ז, ג). שניהם שמחו, ונתן הלך לדרכו. אך מאוחר באותו לילה דיבר אלוהים אל נתן.

הנביא התעורר. עיניו נשואות אל התקרה, ודיבר אל עצמו בחשכה. "בַּבֹּקֶר הָיִיתִי נְבִיא הַמֶּלֶךְ," אמר. "בַּלַּיְלָה — נַעֲשֵׂיתִי נְבִיא ה'." שתק, וידע שאין זה אותו התפקיד. "אָמַרְתִּי לוֹ: 'לֵךְ עֲשֵׂה כָּל אֲשֶׁר בִּלְבָבֶךָ' (שמ"ב ז, ג). שָׂמַח — וַיֵּלֶךְ. אָמַרְתִּי מַה שֶּׁרָצָה לִשְׁמֹעַ. כָּל כָּךְ קַל הָיָה. וּבַלַּיְלָה הַהוּא — הָיָה דְּבַר ה' אֶל נָתָן."

הוא התיישב במיטתו. "נָקֵל לַנָּבִיא לִרְדֹּף אַחַר חֵן הַמֶּלֶךְ," הרהר. "אַךְ הַנָּבִיא אֲשֶׁר יֵלֵךְ אַחַר לִבּוֹ — וְלֹא אַחַר פִּי ה' — הוּא לֹא נָבִיא; הוּא חֹנֵף. וּבָזֶה אֵין כָּבוֹד — אֶלָּא מַפֶּלֶת." הוא נשם, וזכר את הקול. "וְהַקּוֹל אָמַר: 'הַאַתָּה תִּבְנֶה לִי בַּיִת לְשִׁבְתִּי?' (שמ"ב ז, ה) — שְׁאֵלָה שֶׁאֵינָהּ שְׁאֵלָה, תְּשׁוּבָה שֶׁיֵּשׁ בְּתוֹכָהּ."

ובלבו עלה הטעם הכבד שמאחורי הדברים: "'כִּי דַם לָרֹב שָׁפַכְתָּ וּמִלְחָמוֹת גְּדֹלֹת עָשִׂיתָ — לֹא תִבְנֶה בַּיִת לִשְׁמִי' (דה"א כב, ח). הַחֶרֶב — מַה תַּעֲשֶׂה לַנֶּפֶשׁ הַמְּרִימָה אוֹתָהּ?" הוא התחבט בדבר. הלחימה לשם ה' אמת היא; אך הדם אשר שפך — גם שפך. ואין לבית המקדש יסוד מדם.

לבסוף קיבל את ההכרעה. "מָחָר אֹמַר לוֹ," אמר, "'לֹא אַתָּה תִּבְנֶה — כִּי אִם בִּנְךָ הַיֹּצֵא מֵחֲלָצֶיךָ' (שמ"ב ז, יב-יג). הָאֱלֹהִים לָקַח אֹתוֹ מִן הַנָּוֶה — וְלַמִּקְדָּשׁ — בָּחַר בַּבֶּן." חיוך של הרהור עלה בו. "הַמִּקְדָּשׁ אֵינֶנּוּ שֶׁל דָּוִד. הַמִּקְדָּשׁ — שֶׁל אֲשֶׁר יָבוֹא אַחֲרָיו." הוא הביט אל החלון, וסיים בלחש: "כָּבֵד הַדָּבָר — לַנָּבִיא וְגַם לַמֶּלֶךְ. אַךְ 'נֶאֱמָן ה' אֱלֹהִים בְּכָל דְּבָרָיו' (תהלים קמה, יג). וַאֲנִי אוֹמֵר מַה שֶׁנֶּאֱמַר לִי — לֹא מַה שֶׁקַּל לְאָמְרוֹ."

בבוקר בא נתן אל דוד, ושניהם ישבו. הנביא הניח ידו על כתף המלך, ואז נטל אותה, ודיבר. "כֹּה אָמַר ה': 'הַאַתָּה תִּבְנֶה לִי בַּיִת לְשִׁבְתִּי?'" שתק רגע. "'כִּי לֹא יָשַׁבְתִּי בְּבַיִת — מִיּוֹם הַעֲלֹתִי אֶת בְּנֵי יִשְׂרָאֵל מִמִּצְרַיִם — הֲצֹּו צִוִּיתִי לְאַחַד שִׁבְטֵי יִשְׂרָאֵל לִבְנוֹת בֵּית אֶרֶז?'" דוד שמע, ולא הפריע.

נתן המשיך, ונימתו חמה יותר: "'וְעַתָּה כֹּה תֹאמַר לְעַבְדִּי לְדָוִד: אֲנִי לְקַחְתִּיךָ מִן הַנָּוֶה מֵאַחַר הַצֹּאן — לִהְיוֹת נָגִיד עַל עַמִּי יִשְׂרָאֵל. וָאֶהְיֶה עִמְּךָ בְּכֹל אֲשֶׁר הָלַכְתָּ—'"

ודוד מלמל בשקט, כמי שתופס דווקא את המילה הזו ולא את האיסור: "...בְּכֹל אֲשֶׁר הָלַכְתִּי..." ולעצמו: "גַּם בַּמַּעֲרָה. גַּם בְּצִקְלָג. גַּם עַכְשָׁו."

נתן הוסיף: "'כִּי יִמְלְאוּ יָמֶיךָ וְשָׁכַבְתָּ אֶת אֲבֹתֶיךָ — וַהֲקִימֹתִי אֶת זַרְעֲךָ אַחֲרֶיךָ... הוּא יִבְנֶה בַּיִת לִשְׁמִי, וְכֹנַנְתִּי אֶת כִּסֵּא מַמְלַכְתּוֹ עַד עוֹלָם.'"

שתיקה ארוכה ירדה. דוד ישב, ופניו לא נפלו — הוא חשב. ולבסוף דיבר בקול שקט שלא היה בו אכזבה: "בִּקַּשְׁתִּי לָתֵת לוֹ בַּיִת — וְהוּא נוֹתֵן בַּיִת לִי." כמעט חייך. "מִי מֵקִים בַּיִת לְמִי, נָתָן?"

ונתן השיב בשקט: "זוֹ הַשְּׁאֵלָה הַנְּכוֹנָה הָרִאשׁוֹנָה שֶׁשָּׁאַלְתָּ הַיּוֹם."

אחר כך הלך דוד לפני הארון, ושכב על הארץ, ודיבר. לא כעס ולא אכזבה היו בתפילתו, אלא הכנעה. זו הייתה תפילתו האמיתית:

> מִי אָנֹכִי ה' אֱלֹהִים —
> וּמִי בֵיתִי —
> כִּי הֲבִיאֹתַנִי עַד הֲלֹם?!

תמהון אמיתי עלה בו, והוא נזכר מאין בא:

> מֵהַנָּוֶה — מֵאַחַר הַצֹּאן.
> בֶּן שְׁלֹשִׁים.
> לֹא אָבִי, לֹא אָחַי.
> 'אָנֹכִי תוֹלַעַת וְלֹא אִישׁ —
> חֶרְפַּת אָדָם וּבְזוּי עָם.' *(תהלים כב, ז)*
> וְאַתָּה אָמַרְתָּ: מֶלֶךְ.

ומלבו עלה הדבר היחיד שביקש מעודו:

> 'אַחַת שָׁאַלְתִּי מֵאֵת ה' — אוֹתָהּ אֲבַקֵּשׁ —
> שִׁבְתִּי בְּבֵית ה' כָּל יְמֵי חַיַּי!' *(תהלים כז, ד)*
> לֹא מַמְלָכָה. לֹא עֹז.
> רָצִיתִי לָשֶׁבֶת בְּבֵיתְךָ.
> וְאַתָּה נָתַתָּ לִי אֶת הַבַּיִת הָאֱנוֹשִׁי —
> אֶת הַמַּמְלָכָה — לִשְׁמֹר לְךָ.

ואז התרוממה נפשו:

> 'כִּי אַבִּיט שָׁמֶיךָ מַעֲשֵׂה אֶצְבְּעֹתֶיךָ —
> יָרֵחַ וְכוֹכָבִים אֲשֶׁר כּוֹנַנְתָּה —
> מָה אֱנוֹשׁ כִּי תִזְכְּרֶנּוּ?!' *(תהלים ח, ד-ה)*

שתיקה. ואז, בקבלה: "...וַתִּזְכְּרֵנִי." נשם, וקיבל. "בְּנִי יִבְנֶה. טוֹב." והמשיך מתפילתו שלו:

> וְעַתָּה ה' אֱלֹהִים —
> הַדָּבָר אֲשֶׁר דִּבַּרְתָּ עַל עַבְדְּךָ —
> הָקֵם עַד עוֹלָם —
> וַעֲשֵׂה כַּאֲשֶׁר דִּבַּרְתָּ.
> וְיִגְדַּל שִׁמְךָ עַד עוֹלָם!
> *(שמ"ב ז, כה)*

קם דוד מתפילתו, ועבר זמן. ומה שהבין בלבו לפני הארון הבשיל בו עתה למעשה: אם לא הוא יבנה את הבית, יבנה את השיר שיחיה בתוך הבית. החצר נפתחה לפניו, ובה לויים ומשוררים מסודרים — אסף, הימן, ידותון, שלושה ראשי קבוצות. דוד עבר ביניהם.

"אָסָף — הִנָּבֵא!" אמר בשקט. ולידותון: "יְדוּתוּן — הִנָּבֵא!" ולהימן: "הֵימָן — הִנָּבֵא!" והם החלו לנגן — ולא רק לנגן, כי הנגינה עצמה הייתה נבואה, "הַנְּבִאִים בְּכִנֹּרוֹת".

וקול נשמע, מספר את הדבר: "וַיַּבְדֵּל דָּוִד... לַעֲבֹדָה לִבְנֵי אָסָף וְהֵימָן וִידוּתוּן (דה"א כה, א), הַנִּבְּאִים בְּכִנֹּרוֹת בִּנְבָלִים וּבִמְצִלְתָּיִם... כָּל אֵלֶּה בָנִים... עַל יְדֵי אֲבִיהֶם בַּשִּׁיר בֵּית ה'... (דה"א כה, ו) וַיְהִי מִסְפָּרָם... כָּל הַמֵּבִין — מָאתַיִם שְׁמוֹנִים וּשְׁמוֹנָה" (דה"א כה, ז).

דוד עצר, והביט בכולם, ואז דיבר לעצמו, כמו אל ה': "אֶת הַמִּקְדָּשׁ — לֹא אֶבְנֶה. אֶת הָעֲבֹדָה — אָכִין. אֶת הַשִּׁיר — אַנִּיחַ. וּכְשֶׁיִּבְנֶה שְׁלֹמֹה אֶת הַבַּיִת — יִהְיֶה שָׁם שִׁיר שֶׁכְּבָר יָדַע אֶת דַּרְכּוֹ." הוא הביט שוב במשוררים, והם המשיכו לנגן, והנגינה לא נקטעה — היא נמשכה הלאה, אל השיר שיָשִיר לבנו. כאן הבין דוד שמורשתו שירית, לא אדריכלית; ש"נְעִים זְמִרוֹת יִשְׂרָאֵל" אינו תואר שנתן לעצמו, אלא תואר שצמח מן המעשה. עצירת היד מן הבנייה הולידה כינור.

שנים חלפו, ודוד היה לזקן. קרא לשלמה, אחז את ידו, ושר לו — לא כנסיך לנסיך, אלא כאב לבן. כל בית והפסקתו, ובסוף שקט מוחלט. שלמה לא ענה. רק הקשיב.

> אִם אָנֹכִי חָצַבְתִּי אֶבֶן לְהָקִים בַּיִת חָדָשׁ,
> (אִם חָצַבְתִּי אֶבֶן, לְהָקִים בַּיִת חָדָשׁ)
> לֹא לַשָּׁוְא, בְּנִי, חָצַבְתִּי — לֹא לִי יָקוּם הַמִּקְדָּשׁ,
> כִּי מִן הָאֲבָנִים הָאֵלֶּה — יָדְךָ תִּבְנֶה הַמִּקְדָּשׁ —
> יִבָּנֶה, יִבָּנֶה, יִבָּנֶה הַמִּקְדָּשׁ.
>
> אִם אָנֹכִי שָׁפַכְתִּי דָּם לְמַעַן שֶׁקֶט יָעוּן,
> (אִם שָׁפַכְתִּי דָּם — לְמַעַן שֶׁקֶט יָעוּן)
> לֹא לַשָּׁוְא, בְּנִי, לָחַמְתִּי — אִישׁ מְנוּחָה אַתָּה תִכּוֹן,
> כִּי מִן הַקְּרָבוֹת הַלָּלוּ — יָבוֹא שַׁלְוָה וָשֶׁקֶט יָצוּב —
> יָקוּם, יָקוּם, יָקוּם בַּיִת שָׁלוֹם. *(דה"א כב, ח-ט)*
>
> אִם אֲנִי לֹא אֲנִי הַבּוֹנֶה — בְּנִי יִהְיֶה הַבַּנָּאִי,
> (לֹא אֲנִי הַבּוֹנֶה — בְּנִי יִהְיֶה הַבַּנָּאִי)
> אֲנִי הָכִינוֹתִי הַכֹּל, כֶּסֶף וְזָהָב מָשׁוּק,
> אֶת הַנְּחֹשֶׁת וְהָעֵצִים — אַתָּה תַעֲשֶׂה הַפָּסוּק —
> יִבָּנֶה, יִבָּנֶה, יִבָּנֶה הַמִּקְדָּשׁ.
>
> אִם לֹא שַׁרְתִּי שִׁיר עֲדַיִן — אָשִׁיר לִי מִזְמוֹר חָדָשׁ,
> שֶׁהוּא עַתִּיק מִיֵּינָם — וּמָתוֹק מִדְּבַשׁ,
> שִׁיר קַדְמוֹן כִּימֵי עוֹלָם — וּבְכָל יוֹם הוּא חָדָשׁ,
> שִׁיר שֶׁלֹּא אָנֹכִי אָשִׁיר — כִּי אִם שְׁלֹמֹה בְּנִי הַקָּדוֹשׁ —
> יִבָּנֶה, יִבָּנֶה, יִבָּנֶה הַמִּקְדָּשׁ.
>
> כִּי לֹא אָנֹכִי הַבּוֹנֶה — כִּי אִם בְּנִי שַׁלְוָתִי.
> שְׁמוֹ שְׁלֹמֹה, וּשְׁלוֹמוֹ — יַנְחִיל אֶת בֵּית אֲבוֹתַי.
> יִבָּנֶה, יִבָּנֶה, יִבָּנֶה הַמִּקְדָּשׁ.

וקולו של הזקן נמוג, ושלמה עומד לפניו ומקשיב, והשיר — שיר שעוד לא נבנה לו בית — היה כבר מונח ביסודות הבית שיבוא.
