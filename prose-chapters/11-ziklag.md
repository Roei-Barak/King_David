# פֶּרֶק י: צִקְלַג

אֵשׁ הַמַּחֲנֶה דָּעֲכָה בְּקָצֶה הַמִּדְבָּר. אַנְשֵׁי דָּוִד יָשְׁנוּ פְּזוּרִים, מְתֻשִּׁים, וְדָוִד וַאֲבִישַׁי לְבַדָּם נוֹתְרוּ עֵרִים, מַבִּיטִים מַעֲרָבָה — אֶל הָעֲרָפֶל שֶׁמֵּעֵבֶר לַגְּבוּל, אֶל אֶרֶץ פְּלִשְׁתִּים.

אֲבִישַׁי דִּבֵּר רִאשׁוֹן, שָׁקֵט וְנוֹאָשׁ. "שָׁאוּל לֹא יֶחְדַּל, דָּוִד. כָּרַתָּ אֶת כְּנַף מְעִילוֹ וְחָמַל עָלָיו — וּלְמָחֳרָת שׁוּב שָׁלֹשֶׁת אֲלָפִים אֲנָשִׁים בָּחוּר בְּעִקְּבוֹתֵינוּ. אֵין עוֹד סֶלַע בְּכָל יְהוּדָה שֶׁלֹּא יָדַעְנוּ אֶת צִלּוֹ."

דָּוִד הֵשִׁיב לְאַט, וּמְרִירוּת בְּקוֹלוֹ. "עַתָּה אֶסָּפֶה יוֹם אֶחָד בְּיַד שָׁאוּל. אֵין לִי טוֹב כִּי הִמָּלֵט אִמָּלֵט אֶל אֶרֶץ פְּלִשְׁתִּים — וְנוֹאַשׁ מִמֶּנִּי שָׁאוּל לְבַקְשֵׁנִי עוֹד בְּכָל גְּבוּל יִשְׂרָאֵל." (על פי כז, א)

אֲבִישַׁי נִדְהַם, וְקוֹלוֹ נִשְׁבַּר לְחֵצִי לְחִישָׁה. "אֶל גַּת?! אֶל אֶרֶץ הָעָרֵל אֲשֶׁר הֵבֵאתָ עָלָיו אֶת רֹאשׁ עֲנָקוֹ? אֶל עִיר גָּלְיָת תִּסֹּב פָּנֶיךָ — אַתָּה, מְשִׁיחַ הַשֶּׁמֶן?"

עֵינָיו שֶׁל דָּוִד הָיוּ כְּבוּיוֹת, אַךְ דְּבָרָיו הָיוּ קָשִׁים. "הַנִּרְדָּף אֵין לוֹ אֶרֶץ, אֲבִישַׁי. רַק גְּבוּל אֶחָד מֵעֵבֶר לַחֶרֶב. מוּטָב חַיִּים בְּצֵל אוֹיֵב — מִמָּוֶת בְּיַד אֲחִי-עַמִּי."

הֵם קָמוּ, וְהֵעִירוּ אֶת הָאֲנָשִׁים בִּלְחִישָׁה. שֵׁשׁ מֵאוֹת צְלָלִים נָעוּ מַעֲרָבָה, אֶל הָעֲרָפֶל.

*

בְּאוּלָם נָמוּךְ וְזָר יָשַׁב אַכִישׁ מֶלֶךְ גַּת — פְּסָלִים פְּלִשְׁתִּיִּים סָבִיב, וְאֵשׁ שֶׁמֶן מְפַעְפַּעַת בָּעֲשָׁשִׁיּוֹת. דָּוִד עָמַד מוּלוֹ, רֹאשׁוֹ מֻרְכָּן מְעַט — לֹא כְּעֶבֶד, אַךְ לֹא כְּמֶלֶךְ.

"דָּוִד בֶּן-יִשַׁי," אָמַר אַכִישׁ, חֲצִי-לַעַג חֲצִי-עִנְיָן בְּקוֹלוֹ. "הַמַּכֶּה בְּרִבְבוֹתָיו. הַיּוֹם — נָס אֶל שַׁעֲרֵי גַּת. הִקְאִישׁ הִקְאַשְׁתָּ אֶת שָׁאוּל בְּעַמְּךָ — וְעַל כֵּן תִּהְיֶה לִי עֶבֶד עַד עוֹלָם." (על פי כז, יב)

קוֹלוֹ שֶׁל דָּוִד הָיָה נָמוּךְ וּמָדוּד. "אִם נָא מָצָאתִי חֵן בְּעֵינֶיךָ, יִתְּנוּ לִי מָקוֹם בְּאַחַת עָרֵי הַשָּׂדֶה וְאֵשְׁבָה שָּׁם — וְלָמָּה יֵשֵׁב עַבְדְּךָ בְּעִיר הַמַּמְלָכָה עִמָּךְ?" (על פי כז, ה)

אַכִישׁ נוֹפֵף יָדוֹ, נָדִיב מִתּוֹךְ חִשּׁוּב. "תִּהְיֶה לְךָ צִקְלַג." (על פי כז, ו) הוּא נִשְׁעַן לְאָחוֹר וְחִיֵּךְ חִיּוּךְ קָר. "שֵׁב בָּהּ, וּפְשֹׁט אֶת אֲשֶׁר תִּפְשֹׁט — וְהָבֵאתָ לִי מִן הַשָּׁלָל. הֲרֵי כְּבָר הָיִיתָ לְבַז בְּעַמֶּךָ — נֶאֱמָן תִּהְיֶה לִי, כִּי לֹא יִהְיֶה לְךָ אֶל מִי לָשׁוּב."

דָּוִד הִשְׁתַּחֲוָה. פָּנָיו אֲטוּמוֹת. אַךְ כְּשֶׁהִסְתּוֹבֵב, נִרְאָה הַמְּחִיר בְּעֵינָיו.

*

בַּלַּיְלָה עָמַד דָּוִד לְבַדּוֹ עַל חוֹמַת צִקְלַג, וּפָנָיו מִזְרָחָה — אֶל אֶרֶץ יִשְׂרָאֵל שֶׁמֵּעֵבֶר. וּבְלִבּוֹ עָלְתָה תְּפִלָּה אֶל הַחֹשֶׁךְ מִמִּזְרָח:

> שֵׁשׁ מֵאוֹת בָּתֵּי לֶחֶם תְּלוּיִם בִּי, אֱלֹהַי —
> נָשִׁים, וְטַף, וְזָקֵן —
> וַאֲנִי מַאֲכִיל אוֹתָם מִיַּד עוֹרֵל.
> לוֹחֵשׁ לוֹ "כֵּן, אֲדֹנִי" — וּפוֹשֵׁט בַּלַּיְלָה עַל אוֹיְבֵי עַמְּךָ,
> וּמַסְתִּיר אֶת הַדָּם, וּמְשַׁקֵּר אֶת פִּי לְמַעַן יִחְיוּ הַחַיִּים.

הוּא שָׁתַק, וְהִבִּיט בְּיָדָיו.

> זֹאת הִיא תְּפִלַּת הַנִּרְדָּף, רִבּוֹנוֹ שֶׁל עוֹלָם:
> לֹא "תֵּן לִי כֶּתֶר" — אֶלָּא "תֵּן לִי שֶׁלֹּא אֹבַד אֶת נַפְשִׁי בַּדֶּרֶךְ אֵלָיו".
> כִּי קַל הָיָה לִי לִנְפֹּל בַּחֶרֶב —
> קָשֶׁה מִכֹּל לִחְיוֹת, וְלֹא לְהִטָּמֵא עַד הַסּוֹף.

וְהֵרִים מַבָּט אֶל מַעְלָה — לֹא בַּקָּשָׁה, אֶלָּא הַצְהָרָה שְׁקֵטָה שֶׁשָּׁמְרָה עָלָיו:

> ה' אוֹרִי וְיִשְׁעִי מִמִּי אִירָא —
> ה' מָעוֹז חַיַּי מִמִּי אֶפְחָד. (תהילים כז, א)

הָאוֹר דָּעַךְ. מֵעֵבֶר נִשְׁמְעוּ תֻּפִּים נְמוּכִים מִתְקָרְבִים — סִימָן שֶׁל קְרָב, וְשֶׁל זְמַן שֶׁעוֹבֵר.

*

הָיָה זֶה בְּאוֹר עַז וְקָשֶׁה שֶׁשָּׁבוּ דָּוִד וַאֲנָשָׁיו מִן הַדֶּרֶךְ, עֲיֵפִים, נוֹשְׂאֵי נֶשֶׁק. וְאָז קָפְאוּ. מוּלָם עָמְדָה צִקְלַג: עָשָׁן. אֵפֶר. שְׁתִיקָה. אֵין נָשִׁים. אֵין יְלָדִים. אֵין קוֹל. רַק רוּחַ בָּאֵפֶר.

"הָעִיר... שְׂרוּפָה בָּאֵשׁ," לָחַשׁ אֲבִישַׁי בְּקוֹל שָׁבוּר, וְהִסְתּוֹבֵב סָבִיב סָבִיב. "וְהַנָּשִׁים — אַיֵּה? וְהַטַּף — אַיֵּה?"

אִישׁ נָפַל עַל בִּרְכָּיו בָּאֵפֶר. וְהַקּוֹל עָלָה מִקֶּרֶב הָאֲנָשִׁים, נִשְׁבָּר: "שְׁבוּ עֲמָלֵק אֶת הַנָּשִׁים אֲשֶׁר בָּהּ — לֹא הֵמִיתוּ — וַיִּנְהֲגוּ וַיֵּלְכוּ לְדַרְכָּם." (על פי ל, ב)

וְאָז פָּרְצוּ כֻּלָּם בִּבְכִי. בְּכִי גָּדוֹל, מִן הַבֶּטֶן, וַיִּשְׂאוּ קוֹלָם וַיִּבְכּוּ עַד אֲשֶׁר אֵין בָּהֶם כֹּחַ לִבְכּוֹת (ל, ד). וְאַחַר הַבְּכִי בָּאָה דְּמָמָה מְסֻכֶּנֶת.

הַדְּמָמָה שֶׁאַחֲרֵי הַבֶּכִי הָיְתָה גְּרוּעָה מִן הַבֶּכִי. הָאֲנָשִׁים הִתְרוֹמְמוּ אֶחָד-אֶחָד, וְהִבִּיטוּ בְּדָוִד. מְרִירוּתָם חִפְּשָׂה אָשֵׁם, וְיָדַיִם נָעוּ אֶל אֲבָנִים.

"אַתָּה הֵבֵאתָנוּ הֵנָּה, דָּוִד," אָמַר אֶחָד מֵהֶם, מַר וּמְאַיֵּם. "אַתָּה אָמַרְתָּ 'צִקְלַג'. שָׁם אִשְׁתִּי. שָׁם בָּנַי. וְעַתָּה — אֵפֶר."

"אֶחָי — לֹא! לֹא בְּיָדוֹ הַדָּבָר —" נִסָּה אֲבִישַׁי לִבְלֹם, אַךְ קוֹלוֹ נֶחֱלַשׁ.

אַךְ הָעָם הִתְלַהֵט, וַאֲבָנִים בִּידֵיהֶם. "לִסְקֹל אֹתוֹ! לִסְקֹל אֹתוֹ בָּאֲבָנִים! כִּי מָרָה נֶפֶשׁ כָּל הָעָם — אִישׁ עַל בָּנָיו וְעַל בְּנֹתָיו!" (על פי ל, ו) הֵם הִקִּיפוּ אוֹתוֹ. הוּא לְבַדּוֹ בַּמֶּרְכָּז, בְּלִי נֶשֶׁק מוּרָם, אִשְׁתּוֹ אַף-הִיא בַּשְּׁבִי, וְכָל אֲנָשָׁיו — שֶׁהָיוּ בֵּיתוֹ הָאַחֲרוֹן — הָפְכוּ לְאוֹיְבָיו. זֶה הָיָה הַשֵּׁפֶל הַמֻּחְלָט שֶׁל חַיָּיו עַד כֹּה.

*

הַזְּמַן נֶעֱצַר. הָאֲבָנִים נִתְלוּ בָּאֲוִיר. וּמִתּוֹךְ הַשֶּׁבֶר עָלְתָה בְּלִבּוֹ תְּפִלָּה — לֹא נְאוּם אֶל הָאֲנָשִׁים, אֶלָּא זַעֲקָה פְּנִימִית מִן הַקַּרְקָעִית:

> אִבַּדְתִּי הַכֹּל.
> עִיר — אֵין. אִשָּׁה — שְׁבוּיָה. וְעַתָּה גַּם אֵלֶּה, אַחַי, מְבַקְשִׁים נַפְשִׁי.
> מִי נִשְׁאַר לִי, אֱלֹהַי? מִי נִשְׁאַר —

וְכָאן, לָרִאשׁוֹנָה, לֹא הָיָה הַקּוֹל עוֹד תְּפִלָּה. הָיָה הָאֲשָׁמָה. דָּוִד הֵרִים פָּנִים אֶל שָׁמַיִם שׁוֹתְקִים, וְהַשְּׁאֵלָה נֶהְפְּכָה לִזְעָקָה, וְהַקּוֹל נִסְדַּק:

> עַד אָנָה ה' — תִּשְׁכָּחֵנִי נֶצַח?!
> עַד אָנָה — תַּסְתִּיר אֶת פָּנֶיךָ מִמֶּנִּי?! (תהילים יג, ב)
> עַד אָנָה אָשִׁית עֵצוֹת בְּנַפְשִׁי — יָגוֹן בִּלְבָבִי יוֹמָם?
> עַד אָנָה — יָרוּם אֹיְבִי עָלָי?! (תהילים יג, ג)

הוּא עָצַר. כִּי אֵין מַעֲנֶה. הַשָּׁמַיִם שָׁתְקוּ — כְּשֵׁם שֶׁשָּׁתְקוּ לְשָׁאוּל. הָאֲבָנִים עוֹד תְּלוּיוֹת בָּאֲוִיר. רֶגַע אָרֹךְ וּמְסֻכָּן בֶּאֱמֶת, שֶׁבּוֹ שׁוּם יְשׁוּעָה לֹא יָרְדָה, שׁוּם קוֹל לֹא עָנָה, וְאֵין לוֹ אַף סִבָּה אַחַת לְהַאֲמִין. דָּוִד הִבִּיט בָּאֵפֶר, בְּיָדָיו הָרֵיקוֹת, בַּאֲנָשָׁיו הָאוֹחֲזִים אֲבָנִים. כָּל הַהִגָּיוֹן אָמַר: דַּי.

וּמִתּוֹךְ הַשְּׁתִיקָה עַצְמָהּ — לֹא מִתּוֹךְ מַעֲנֶה, כִּי אֵין מַעֲנֶה — בָּחַר דָּוִד. בְּלַחַשׁ, כִּמְעַט בְּלִי קוֹל:

> הָאִירָה עֵינַי — פֶּן אִישַׁן הַמָּוֶת. (תהילים יג, ד)

שְׁתִיקָה אַחֲרוֹנָה. וְאָז — לֹא כִּי נַעֲנָה, אֶלָּא כִּי בָּחַר לִבְטֹחַ בְּלֹא הַבְטָחָה:

> וַאֲנִי — בְּחַסְדְּךָ בָטַחְתִּי. (תהילים יג, ו)
> מִי נִשְׁאַר לִי?
> אַתָּה. רַק אַתָּה נִשְׁאַרְתָּ.

וְכָאן הִתְרוֹמֵם הַפָּסוּק שֶׁהוּא צִיר כָּל חַיָּיו, הִתְרוֹמֵם מִתּוֹכוֹ, וְהוּא הִתְרוֹמֵם עִמּוֹ מִן הַקַּרְקַע: וַיִּתְחַזֵּק דָּוִד (ל, ו) — בַּה' אֱלֹהָיו.

לֹא יְשׁוּעָה יָרְדָה. דָּוִד הִתְחַזֵּק. לֹא כִּי הַמַּצָּב הִשְׁתַּנָּה — כִּי הוּא בָּחַר. וְעַתָּה, כְּשֶׁהוּא מוּצָק, קָם וּפָנָה אֶל אֶבְיָתָר הַכֹּהֵן, וְקוֹלוֹ הִתְיַצֵּב: "הַגִּישָׁה נָּא לִי הָאֵפוֹד, אֶבְיָתָר." (על פי ל, ז)

אֶבְיָתָר הִגִּישׁ אֶת הָאֵפוֹד. דָּוִד שָׁאַל, וְהַדְּמָמָה מְתוּחָה, וְהָאֲבָנִים עוֹד בְּיַד הָאֲנָשִׁים. "הַאֶרְדֹּף אַחֲרֵי הַגְּדוּד הַזֶּה — הַאַשִּׂגֶנּוּ?"

רֶגַע. וְאָז נָשָׂא קוֹלוֹ אֶת תְּשׁוּבַת ה' אֶל הָעָם, וְהֶעֱבִיר אֲלֵיהֶם אֶת הַבִּטָּחוֹן: "רְדֹף — כִּי הַשֵּׂג תַּשִּׂיג, וְהַצֵּל תַּצִּיל!" (על פי ל, ח)

הָאֲבָנִים יָרְדוּ מִן הַיָּדַיִם, אַחַת-אַחַת. הַקּוֹל שֶׁהָיָה זַעַם נֶהְפַּךְ לְתִקְוָה. אֲבִישַׁי הָרִאשׁוֹן הֵרִים נֶשֶׁק — לֹא נֶגֶד דָּוִד, אֶלָּא אַחֲרָיו. "אַחֲרֵי דָּוִד! רְדֹף — וְהַצֵּל!" עָלָה קוֹל הָעָם.

*

הָיְתָה תְּנוּעָה, מָצוֹד, תֻּפִּים. שֵׁשׁ מֵאוֹת יָצְאוּ, וּמָאתַיִם נֶחֱלְשׁוּ וְיָשְׁבוּ עַל הַכֵּלִים אֵצֶל נַחַל הַבְּשׂוֹר. וּבְמַהֲלַךְ הַמִּרְדָּף מָצְאוּ נַעַר מִצְרִי מֻטָּל בַּשָּׂדֶה, גּוֹוֵעַ.

"אִישׁ מִצְרִי בַּשָּׂדֶה, נָטוּשׁ, רָעֵב," הִצְבִּיעַ אֲבִישַׁי.

דָּוִד נָתַן לוֹ לֶחֶם וּמַיִם, רַךְ אַף בְּעִצּוּמוֹ שֶׁל הַמָּצוֹד. "מִי אַתָּה, וּלְמִי אַתָּה?" (על פי ל, יג)

חַלָּשׁ הֵשִׁיב הַנַּעַר: "נַעַר מִצְרִי אָנֹכִי, עֶבֶד לְאִישׁ עֲמָלֵקִי — וַיַּעַזְבֵנִי אֲדֹנִי כִּי חָלִיתִי הַיּוֹם שְׁלֹשָׁה. אֲנַחְנוּ פָּשַׁטְנוּ... וְאֶת צִקְלַג שָׂרַפְנוּ בָּאֵשׁ. (על פי ל, יג–יד) אִם תִּשָּׁבַע לִי לֹא תְמִיתֵנִי — וְאוֹרִידְךָ אֶל הַגְּדוּד."

"הוֹרִידֵנִי," אָמַר דָּוִד.

הַקְּרָב לֹא נִרְאָה בְּדָם, אֶלָּא בִּתְנוּעָה וּבְאוֹר. וְאָז הִתְבַּהֵר הָאוֹר. נָשִׁים רָצוּ אֶל בַּעֲלֵיהֶן, יְלָדִים אֶל אֲבוֹתֵיהֶם, וְדָוִד אָחַז בְּיָדוֹ — סִימָן הֲשָׁבַת אִשְׁתּוֹ. אֲבִישַׁי נָשַׁם, וְלֹא הֶאֱמִין: "וְלֹא נֶעְדַּר לָהֶם מִן הַקָּטֹן וְעַד הַגָּדוֹל, מִבָּנִים וּבָנוֹת וּמִשָּׁלָל — הַכֹּל הֵשִׁיב דָּוִד." (על פי ל, יט)

*

שָׁבוּ אֶל נַחַל הַבְּשׂוֹר, אֶל הַמָּאתַיִם שֶׁנּוֹתְרוּ עִם הַכֵּלִים. וְאַנְשֵׁי הַקְּרָב, אַנְשֵׁי בְּלִיַּעַל, הִתְנַגְּדוּ לַחֲלֹק עִמָּם. אֶחָד מֵהֶם דִּבֵּר, קָשֶׁה וְצַר-עַיִן: "יַעַן אֲשֶׁר לֹא הָלְכוּ עִמָּנוּ — לֹא נִתֵּן לָהֶם מֵהַשָּׁלָל אֲשֶׁר הִצַּלְנוּ! אִישׁ אֶת אִשְׁתּוֹ וְאֶת בָּנָיו — וְיֵלֵכוּ." (על פי ל, כב)

דָּוִד פָּסַע אֶל הַמֶּרְכָּז, וְקוֹלוֹ הָיָה קוֹל מֶלֶךְ, חוֹתֵךְ. "לֹא תַעֲשׂוּ כֵן, אֶחָי, אֵת אֲשֶׁר נָתַן ה' לָנוּ — וַיִּשְׁמֹר אֹתָנוּ, וַיִּתֵּן אֶת הַגְּדוּד הַבָּא עָלֵינוּ בְּיָדֵנוּ. (על פי ל, כג) לֹא בְּכֹחֵנוּ נִצַּחְנוּ — כִּי אִם בְּיָדוֹ. וְהַשָּׁלָל — שֶׁלּוֹ הוּא, נָתוּן לְכֻלָּנוּ."

וּפָנָה אֶל כָּל הָעָם, וְהֵרִים קוֹלוֹ — זֶה הָיָה הָרֶגַע שֶׁבּוֹ הָפַךְ הַנִּרְדָּף, לָרִאשׁוֹנָה, לְמֶלֶךְ: "כִּי כְּחֵלֶק הַיֹּרֵד בַּמִּלְחָמָה — וּכְחֵלֶק הַיֹּשֵׁב עַל הַכֵּלִים — יַחְדָּו יַחֲלֹקוּ!" (ל, כד)

דְּמָמָה. וְאָז קַבָּלָה. גַּם הַמָּאתַיִם נִכְלְלוּ. לֹא רַק חֲלֻקַּת שָׁלָל הָיְתָה זוֹ, אֶלָּא חָזוֹן שֶׁל עַם שֶׁאִישׁ בּוֹ אֵינוֹ נִשְׁאָר מֵאָחוֹר. כָּאן נוֹלַד חֹק וּמִשְׁפָּט שֶׁיִּלְוֶה אֶת דָּוִד, וַיְשִׂמֶהָ לְחֹק וּלְמִשְׁפָּט לְיִשְׂרָאֵל עַד הַיּוֹם הַזֶּה (ל, כה).

*

עֶרֶב. מְדוּרוֹת צִקְלַג בִּעְבְּעוּ מֵחָדָשׁ — לֹא אֵשׁ חֻרְבָּן, אֶלָּא אֵשׁ בַּיִת. נָשִׁים, יְלָדִים, לוֹחֲמִים יַחַד. וְדָוִד יָשַׁב, וְלָרִאשׁוֹנָה זֶה זְמַן רַב — שָׁלֵם. וְקוֹלוֹ עָלָה בַּשִּׁיר, וְהָעָם עָנָה אַחֲרָיו:

> בָּעֶרֶב יָלִין הַבֶּכִי וְהַכְּאֵב,
> וְלַבֹּקֶר רִנָּה תָּשׁוּב אֶל הַלֵּב. (על פי תהילים ל, ה)
> כִּי רֶגַע בְּאַפּוֹ — וְחַיִּים בִּרְצוֹנוֹ,
> מִי שֶׁנּוֹפֵל — בְּיָדוֹ יַעֲמֹד עַל מְכוֹנוֹ.
>
> *(וְכָל הָעָם בַּפִּזְמוֹן:)*
> הִתְחַזֵּק! הִתְחַזֵּק בַּה' אֱלֹהֶיךָ,
> כְּשֶׁכָּל הָעוֹלָם נִשְׁבָּר סְבִיבֶךָ.
> לֹא הַיְשׁוּעָה הִיא הַבּוֹחֶרֶת בָּנוּ —
> אֲנַחְנוּ בּוֹחֲרִים לְהַאֲמִין בְּתוֹךְ הַחֹשֶׁךְ אֵלָיו.

וּבְשֶׁקֶט, אִישִׁי, שָׁר דָּוִד אֶת בֵּיתוֹ:

> אִבַּדְתִּי עִיר, וְאִשָּׁה, וּבֵיתָה —
> וְשַׁבְתִּי וּמָצָאתִי אֶת רֹב אַהֲבָתָהּ.
> לֹא כִּי הָיִיתִי חָזָק מֵאֲחֵרִים —
> אַךְ נִשְׁעַנְתִּי עַל מִי שֶׁמֵּקִים נוֹפְלִים.

וְכָל הָעָם הֵשִׁיב בַּפִּזְמוֹן, שִׂיא:

> הִתְחַזֵּק! הִתְחַזֵּק בַּה' אֱלֹהֶיךָ —
> בָּעֶרֶב דְּמָעָה, וּבַבֹּקֶר — בְּרָכָה.

וְדָוִד הֵרִים מַבָּט אֶל מַעְלָה, וְחָתַם בְּשׁוּרַת הַמִּזְמוֹר, שְׁקֵטָה, כִּמְעַט אֶל עַצְמוֹ:

> לוּלֵא הֶאֱמַנְתִּי לִרְאוֹת בְּטוּב ה'
> בְּאֶרֶץ חַיִּים... (תהילים כז, יג)

הוּא לֹא סִיֵּם אֶת הַמִּשְׁפָּט — כְּמָסֹרֶת הַפָּסוּק הַנִּפְסָק. הָאוֹר עָלָה אַט-אַט אֶל הַבֹּקֶר.
