# פֶּרֶק ו: יְמֵי הַמִּדְבָּר

בַּחֲצַר אַרְמוֹן אָכִישׁ מֶלֶךְ גַּת עָמַד אוֹר קָשֶׁה וְעוֹיֵן. דָּוִד נִכְנַס וּבִגְדּוֹ הַיִּשְׂרְאֵלִי עָלָיו, מֻכָּר לְכֹל, וְעַבְדֵי אָכִישׁ הֵחֵלּוּ לוֹחֲשִׁים בְּזִלְזוּל. הֲלוֹא זֶה דָוִד מֶלֶךְ הָאָרֶץ, אָמְרוּ; הֲלוֹא לָזֶה יַעֲנוּ בַמְּחֹלוֹת לֵאמֹר — הִכָּה שָׁאוּל בַּאֲלָפָיו, וְדָוִד בְּרִבְבֹתָיו?

דָּוִד שָׁמַע, וְעָצַר. עֵינָיו מַחְשְׁבוֹת, וְהוּא הֵבִין: אִם יַכִּירוּהוּ, מֵת הוּא. וְעָלוּ בְּלִבּוֹ מִלּוֹת הַמִּזְמוֹר, אֲבָרֲכָה אֶת ה' בְּכָל עֵת — וְאָז בָּחַר. בְּבַת אַחַת נֶהֶפְכוּ פָּנָיו. הוּא הֵחֵל לְהִתְנַהֵג כִּמְשֻׁגָּע, מְסַמֵּן עַל דַּלְתוֹת הַשַּׁעַר וּמוֹרִיד רִיר עַל זְקָנוֹ. וְאָכִישׁ הִתְקָרֵב, מְגֹעָל, וְקָרָא: רְאוּ אִישׁ מְשֻׁגָּע! לָמָּה תָּבִיאוּ אֹתוֹ אֵלָי? הַחֲסַר מְשֻׁגָּעִים אָנִי, כִּי הֲבֵאתֶם אֶת זֶה לְהִשְׁתַּגֵּעַ עָלַי?

נִפְנֵף בְּיָדוֹ — הוֹצִיאוּהוּ — וְדָוִד מָצָא אֶת עַצְמוֹ בַּחוּץ. וּבָרֶגַע שֶׁעָבַר אֶת הַשַּׁעַר, נָפַל הַכֹּל מֵעָלָיו. הוּא הִבִּיט בְּיָדָיו: לֹא רִיר רָאָה עֲלֵיהֶן, כִּי אִם כְּלִמַּת גָּאוֹן. בִּנְשִׁימָה קְצָרָה לָחַשׁ לְעַצְמוֹ: אֲבָרְכָה אֶת ה' בְּכָל עֵת, תָּמִיד תְּהִלָּתוֹ בְּפִי; בַּה' תִּתְהַלֵּל נַפְשִׁי, יִשְׁמְעוּ עֲנָוִים וְיִשְׂמָחוּ. שָׁתַק. הוּא הֵבִין: כְּדֵי לִשְׂרֹד, בִּזָּה אֶת עַצְמוֹ — וַה' הִצִּיל. זֶה הָיָה הַשִּׁעוּר.

הַרְחֵק מִשָּׁם, בָּרָמָה, יָשַׁב שָׁאוּל תַּחַת הָאֵשֶׁל, חֲנִית בְּיָדוֹ וְשָׂרָיו סְבִיבוֹ, וְדוֹאֵג הָאֲדוֹמִי עוֹמֵד בַּצַּד וּמַמְתִּין. בְּכַעַס פָּנָה הַמֶּלֶךְ אֶל שָׂרָיו: שִׁמְעוּ נָא בְּנֵי יְמִינִי! הֲגַם לְכֻלְּכֶם יִתֵּן בֶּן יִשַׁי שָׂדוֹת וּכְרָמִים, כִּי קְשַׁרְתֶּם כֻּלְּכֶם עָלַי וְאֵין מְגֹלֶה אֶת אָזְנִי? שָׁתְקוּ הַשָּׂרִים מִיִּרְאָה. וְאָז דִּבֵּר דּוֹאֵג, שָׁקֵט וּמְחֻשָּׁב: רָאִיתִי אֶת בֶּן יִשַׁי בָּא נֹבָה אֶל אֲחִימֶלֶךְ בֶּן אֲחִטוּב, וַיִּשְׁאַל לוֹ בַּה', וְצֵידָה נָתַן לוֹ, וְאֵת חֶרֶב גָּלְיָת הַפְּלִשְׁתִּי נָתַן לוֹ.

שָׁאוּל הִסְתּוֹבֵב לְאִטּוֹ, וְעֵינָיו דּוֹלְקוֹת. קַר הָיָה קוֹלוֹ: שְׁלַח וְקַח אֶת אֲחִימֶלֶךְ בֶּן אֲחִטוּב וְאֵת כָּל בֵּית אָבִיו הַכֹּהֲנִים אֲשֶׁר בְּנֹב.

וְהָיוּ הַכֹּהֲנִים עוֹמְדִים, וַאֲחִימֶלֶךְ אָמַר בְּפַחַד: לָמָּה יַחֲרֶה אַף אֲדֹנִי בְּעַבְדּוֹ? עַבְדְּךָ לֹא יָדַע בְּכָל זֹאת דָּבָר, כִּי לֹא יָדַע עַבְדְּךָ דָּבָר גָּדוֹל אוֹ קָטֹן. אַךְ שָׁאוּל פָּנָה אֶל דּוֹאֵג וְצִוָּה: הַסֵּב וּפְגַע בַּכֹּהֲנִים! הַשָּׂרִים לֹא זָזוּ — לֹא הָיוּ מְסֻגָּלִים. דּוֹאֵג זָז. וַיַּכֶּה דֹּאֵג שְׁמֹנִים וַחֲמִשָּׁה אִישׁ נֹשֵׂא אֵפוֹד בָּד, וְאֵת נֹב עִיר הַכֹּהֲנִים הִכָּה לְפִי חָרֶב.

בִּמְעָרַת עֲדֻלָּם נִכְנַס אֶבְיָתָר יָחִיד, בָּהוּל, פָּנָיו חֲרוּשִׁים, וּבִנְשִׁימָה אַחַת אָמַר: שָׁאוּל הָרַג אֶת כֹּהֲנֵי ה'. דָּוִד עָצַר. לֹא שָׁאַל — יָדַע. לְאַט, בִּכְאֵב, כְּמִי שֶׁהַדְּבָרִים נֶחְצָבִים מִתּוֹכוֹ, אָמַר: יָדַעְתִּי בַּיּוֹם הַהוּא כִּי שָׁם דּוֹאֵג הָאֲדוֹמִי, כִּי הַגֵּד יַגִּיד לְשָׁאוּל. אָנֹכִי סַבֹּתִי בְּכָל נֶפֶשׁ בֵּית אָבִיךָ. שָׁהָה בִּדְבָרָיו, כִּי הַדָּם עָלָיו, וְהוּא יוֹדֵעַ.

דְּמָמָה אֲרֻכָּה. דָּוִד עָצַם עֵינָיו, וְאַחַר דִּבֵּר אֶל אֶבְיָתָר בְּשֶׁקֶט, קוֹלוֹ נָמוּךְ אַךְ מוּצָק כְּסֶלַע: שְׁבָה אִתִּי, אַל תִּירָא. ה' אֱלֹהַי בְּךָ חָסִיתִי, הוֹשִׁיעֵנִי מִכָּל רֹדְפָי. כִּי אֲשֶׁר יְבַקֵּשׁ אֶת נַפְשִׁי, יְבַקֵּשׁ אֶת נַפְשֶׁךָ — כִּי מִשְׁמֶרֶת אַתָּה עִמָּדִי. זֶה הָיָה הָרֶגַע שֶׁבּוֹ הִרְגִּישׁ דָּוִד אַשְׁמָה רִאשׁוֹנָה. לֹא יָדַע שֶׁיִּקְרֶה, אַךְ יָדַע שֶׁדּוֹאֵג שָׁם. הוּא נָשָׂא אֶת זֹאת — כִּי עֲוֹנֹתַי עָבְרוּ רֹאשִׁי, כְּמַשָּׂא כָבֵד יִכְבְּדוּ מִמֶּנִּי — אַךְ לֹא הִרְפָּה.

בְּמִדְבַּר זִיף, בְּלֵיל אֶחָד, יָשְׁבוּ שְׁנֵי אֲנָשִׁים עַל אֶבֶן, וּשְׁנֵיהֶם יָדְעוּ כִּי זוֹ הַפַּעַם הָאַחֲרוֹנָה. דָּוִד הִסְתַּתֵּר, וִיהוֹנָתָן הִגִּיעַ לְבַדּוֹ, חִפֵּשׂ בֵּין הַסְּלָעִים, וּמְצָאוֹ. דָּוִד, קָרָא. דָּוִד קָם: יְהוֹנָתָן. הִתְחַבְּקוּ, וְאַחַר יָשְׁבוּ, וְלֹא מִהֲרוּ לְדַבֵּר. לְבַסּוֹף אָמַר יְהוֹנָתָן: אַל תִּירָא, כִּי לֹא תִמְצָאֲךָ יַד שָׁאוּל אָבִי, וְאַתָּה תִּמְלֹךְ עַל יִשְׂרָאֵל. עָצַר, וְהוֹסִיף: וַאֲנִי אֶהְיֶה לְךָ לְמִשְׁנֶה.

דָּוִד הִבִּיט בּוֹ. שְׁנֵיהֶם יָדְעוּ שֶׁזֶּה לֹא יִהְיֶה, וִיהוֹנָתָן יָדַע. יְהוֹנָתָן, אָמַר דָּוִד בְּשֶׁקֶט. הִנִּיחַ יְהוֹנָתָן יָד עַל שִׁכְמוֹ: וַיִּכְרְתוּ שְׁנֵיהֶם בְּרִית לִפְנֵי ה' — הַבְּרִית הַזֹּאת לֹא תָמוּת. שְׁתִיקָה אֲרֻכָּה יָרְדָה, וְהָרוּחַ נָשְׁבָה, וְדָוִד הִבִּיט בּוֹ וּמִלְמֵל בְּלִבּוֹ: וְלִקְדוֹשִׁים אֲשֶׁר בָּאָרֶץ הֵמָּה, וְאַדִּירֵי כָּל חֶפְצִי בָם. וְאָז קָם יְהוֹנָתָן. לֵךְ לְשָׁלוֹם, אָמַר, וְהָלַךְ. דָּוִד עָמַד וְהִבִּיט אַחֲרָיו, וְלֹא קָרָא. יָדַע. וּבְלִבּוֹ עָלָה הַתַּחֲנוּן: חָנֵּנִי אֱלֹהִים חָנֵּנִי, כִּי בְךָ חָסָיָה נַפְשִׁי, וּבְצֵל כְּנָפֶיךָ אֶחְסֶה עַד יַעֲבֹר הַוּוֹת.

וּבָא לֵיל שַׁבָּת בַּמִּדְבָּר. מַחֲנֵה דָוִד בֵּין הַסְּלָעִים, וּמְנוֹרַת שֶׁמֶן יְחִידָה — כֶּתֶם אוֹר קָטָן בְּלֵב הָאֲפֵלָה. אֲנָשָׁיו פְּזוּרִים, עֲיֵפִים עַד עָפָר, אִישׁ עַל נִשְׁקוֹ. אֵין בַּיִת, אֵין שֻׁלְחָן, אֵין נֵר שֶׁל כָּבוֹד — וּבְכָל זֹאת, דָּוִד הִבְחִין בַּשְּׁקִיעָה הָאַחֲרוֹנָה, וְדָבָר מָה נָח עָלָיו. חֲרִישִׁית פָּנָה אֶל אֲנָשָׁיו, לֹא כִּפְקֻדָּה אֶלָּא כִּבְשׂוֹרָה: הִנֵּה בָּאָה הַשַּׁבָּת, גַּם פֹּה, בֵּין הַסְּלָעִים, בְּלִי גַּג וּבְלִי שֻׁלְחָן. לֹא יִקְּחוּ מִמֶּנּוּ אֶת הַיּוֹם הַזֶּה — כִּי הַמָּנוֹחַ אֵינֶנּוּ בַּקִּיר, כִּי אִם בַּלֵּב.

שָׁתַק. אֶחָד הָאֲנָשִׁים הֵרִים רֹאשׁ. דָּוִד הִבִּיט מַעְלָה, וְהַפַּחַד שֶׁאָכַל בּוֹ כָּל הַשָּׁבוּעַ נָסוֹג צַעַד אֶחָד. וּמִתּוֹךְ הַמְּצוּקָה, שֶׁלֹּא נִכְנְעָה, עָלָה הַשִּׁיר. דָּוִד הֵחֵל לָשִׁיר, רַךְ, וַאֲנָשָׁיו הִצְטָרְפוּ לְאִטָּם, קוֹל אַחַר קוֹל, עַד שֶׁכָּל הַמַּחֲנֶה הָפַךְ לְמַקְהֵלָה אַחַת קְטַנָּה בְּלֵב הַמִּדְבָּר:

> מִזְמוֹר שִׁיר לְיוֹם הַשַּׁבָּת.
> טוֹב לְהֹדוֹת לַה' —
> וּלְזַמֵּר לְשִׁמְךָ עֶלְיוֹן.
> לְהַגִּיד בַּבֹּקֶר חַסְדֶּךָ —
> וֶאֱמוּנָתְךָ בַּלֵּילוֹת.

וַאֲנָשָׁיו, לְרֶגַע, חָדְלוּ לִשְׁמֹעַ אֶת הַפַּחַד. מַשֶּׁהוּ מַרְגִּיעַ יָרַד עַל הַמַּחֲנֶה — לֹא נִצָּחוֹן, לֹא יְשׁוּעָה, אֶלָּא מְנוּחָה. כֶּתֶם הָאוֹר הַקָּטָן נִרְאָה לְפֶתַע גָּדוֹל יוֹתֵר. וְדָוִד אָמַר אֶל הַחֹשֶׁךְ, לְעַצְמוֹ: בַּלֵּילוֹת — אֱמוּנָתְךָ. דַּוְקָא בַּלַּיְלָה, דַּוְקָא בַּמִּדְבָּר. מָצָאתִי יוֹם אֶחָד שֶׁאֵינֶנּוּ שֶׁל הַמִּרְדָּף, יוֹם שֶׁכֻּלּוֹ שֶׁלְּךָ. וְאִם יֵשׁ שַׁבָּת — יֵשׁ עוֹד תִּקְוָה. הָאוֹר נִשְׁאַר דּוֹלֵק, וְדָוִד עָצַם עֵינָיו, וּבַפַּעַם הָרִאשׁוֹנָה זֶה שָׁבוּעוֹת — נָחָה דַּעְתּוֹ.

וַיָּמָת שְׁמוּאֵל, וַיִּקָּבֵר בְּבֵיתוֹ בָּרָמָה; וְדָוִד וַאֲנָשָׁיו בְּמִדְבַּר מָעוֹן שָׁמְעוּ. שָׁלַח דָּוִד עֲשָׂרָה נְעָרִים אֶל נָבָל הַכַּרְמְלִי, וַיֹּאמְרוּ: שָׁלוֹם לְךָ וְשָׁלוֹם לְבֵיתֶךָ; שָׁמַעְנוּ כִּי גֹזְזִים לָךְ. כָּל הַיָּמִים אֲשֶׁר הָיִינוּ עִם רֹעֶיךָ לֹא הֶחְסַרְנוּם דָּבָר; וְעַתָּה תֵּן נָא לַעֲבָדֶיךָ. אַךְ נָבָל — שָׁמֵן וּמְלֻבָּשׁ הֵיטֵב, כּוֹס יַיִן בְּיָדוֹ — הִסְתּוֹבֵב בְּבוּז וְאָמַר: מִי דָוִד? וּמִי בֶן יִשָׁי? הַיּוֹם רַבּוּ עֲבָדִים הַמִּתְפָּרְצִים אִישׁ מִפְּנֵי אֲדֹנָיו! וְלָקַחְתִּי אֶת לַחְמִי וְאֶת מֵימַי, וְנָתַתִּי לַאֲנָשִׁים אֲשֶׁר לֹא יָדַעְתִּי אֵי מִזֶּה הֵם?

נִפְנֵף בְּיָדוֹ — לְכוּ — וְהַשְּׁלִיחִים חָזְרוּ וְסִפְּרוּ. פְּנֵי דָוִד הִתְכַּוְּצוּ. חִגְרוּ אִישׁ אֶת חַרְבּוֹ! קָרָא אֶל אֲנָשָׁיו, וּבַדֶּרֶךְ אָמַר: אַךְ לַשֶּׁקֶר שָׁמַרְתִּי אֶת כָּל אֲשֶׁר לָזֶה בַּמִּדְבָּר, וְהֵשִׁיב לִי רָעָה תַּחַת טוֹבָה. כֹּה יַעֲשֶׂה אֱלֹהִים לְאֹיְבֵי דָוִד אִם אַשְׁאִיר מִכָּל אֲשֶׁר לוֹ עַד הַבֹּקֶר מַשְׁתִּין בְּקִיר.

וְאָז בָּאָה אֲבִיגַיִל, טוֹבַת שֶׂכֶל וִיפַת תֹּאַר, רוֹכֶבֶת עַל חֲמוֹר. יָרְדָה וְהִשְׁתַּחֲוְתָה לְדָוִד, פָּנֶיהָ לָאָרֶץ: בִּי אֲנִי אֲדֹנִי הֶעָוֺן! וּתְדַבֶּר נָא אֲמָתְךָ בְּאָזְנֶיךָ, וּשְׁמַע אֵת דִּבְרֵי אֲמָתֶךָ. דָּוִד עָצַר. וְתוֹךְ כְּדֵי נְתִינַת הַמִּנְחָה דִּבְּרָה: אַל יָשִׂים אֲדֹנִי אֶת לִבּוֹ אֶל אִישׁ הַבְּלִיַּעַל הַזֶּה, כִּי כִשְׁמוֹ כֵּן הוּא, נָבָל שְׁמוֹ וּנְבָלָה עִמּוֹ. כִּי יַעֲשֶׂה ה' לַאדֹנִי בַּיִת נֶאֱמָן, כִּי מִלְחֲמוֹת ה' אֲדֹנִי נִלְחָם, וְרָעָה לֹא תִמָּצֵא בְךָ מִיָּמֶיךָ. וְהָיְתָה נֶפֶשׁ אֲדֹנִי צְרוּרָה בִּצְרוֹר הַחַיִּים אֵת ה' אֱלֹהֶיךָ.

דָּוִד הִבִּיט בָּהּ נִפְעָם. הִיא אָמְרָה לוֹ מָה הוּא, בְּטֶרֶם הוּא עַצְמוֹ יָדַע. בְּנַחַת הֵשִׁיב: בָּרוּךְ ה' אֱלֹהֵי יִשְׂרָאֵל אֲשֶׁר שְׁלָחֵךְ הַיּוֹם הַזֶּה לִקְרָאתִי, וּבָרוּךְ טַעְמֵךְ, וּבְרוּכָה אָתְּ אֲשֶׁר כְּלִיתִנִי הַיּוֹם הַזֶּה מִבּוֹא בְדָמִים. הִסְתּוֹבֵב אֶל אֲנָשָׁיו: חֲזֹרוּ. אֲבִיגַיִל פָּחֲדָה, וְאָז רָאֲתָה כִּי הוּא מִתְרַחֵק, וְנָשְׁמָה לִרְוָחָה.

וַיַּךְ ה' אֶת נָבָל וַיָּמֹת. וַיִּשְׁמַע דָּוִד כִּי מֵת נָבָל, וַיֹּאמֶר: בָּרוּךְ ה' אֲשֶׁר רָב אֶת רִיב חֶרְפָּתִי מִיַּד נָבָל.
