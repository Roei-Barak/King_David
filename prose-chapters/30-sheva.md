# פֶּרֶק כט: מֶרֶד שֶׁבַע בֶּן בִּכְרִי

הָאָבָק עוֹד תָּלוּי בָּאֲוִיר מִמִּלְחֶמֶת הָאַחִים, וּבַשּׁוּק נִפְגְּשׁוּ מַבָּטִים עוֹיְנִים בֵּין אַנְשֵׁי יְהוּדָה לְאַנְשֵׁי יִשְׂרָאֵל. וּבְקִדְמַת הַקָּהָל עָמַד שֶׁבַע בֶּן בִּכְרִי, אִישׁ בְּלִיָּעַל בְּעֵינֵי הַכָּתוּב — אַךְ בְּעֵינֵי הַהָמוֹן אִישׁ חַם, מְלֻטָּשׁ, הַיּוֹדֵעַ לִגַּע בַּפֶּצַע. בְּיָדוֹ שׁוֹפָר.

קוֹלוֹ עָלָה מֵצִית אֶת הַטִּינָה, חוֹזֵר וּמַכֶּה כְּפַטִּישׁ: "שִׁמְעוּנִי, יִשְׂרָאֵל! הֶחָלַק לָנוּ הַמֶּלֶךְ כְּחֵלֶק? הֶחָשַׁב אוֹתָנוּ בְּמִנְיַן בָּנָיו? אִישׁ יְהוּדָה לְקָחוּהוּ לָהֶם, וַאֲנַחְנוּ — לֹא נֶחְשַׁבְנוּ!"

וַיִּתְקַע בַּשּׁוֹפָר. קוֹל חַד, קוֹרֵעַ, וְהֶהָמוֹן נִדְרַךְ. וְאָז זָרַק אֶת הַקְּרִיאָה הַגְּדוֹלָה, אֶת הַסִּיסְמָה אֲשֶׁר תִּסְחַף אַחֲרֶיהָ אֶת הַשְּׁבָטִים:

"אֵין לָנוּ חֵלֶק בְּדָוִד, וְלֹא נַחֲלָה לָנוּ בְּבֶן יִשַׁי — אִישׁ לְאֹהָלָיו, יִשְׂרָאֵל!" (שמ"ב כ, א)

וְקָלְטוּ אַנְשֵׁי יִשְׂרָאֵל אֶת הַקְּרִיאָה, וְחָזְרוּ עָלֶיהָ כְּגַל, וּפָנוּ מֵעַל דָּוִד. "אִישׁ לְאֹהָלָיו... אִישׁ לְאֹהָלָיו..." הִתְרוֹקֵן הַשּׁוּק מֵאַנְשֵׁי יִשְׂרָאֵל, וְרַק אַנְשֵׁי יְהוּדָה נוֹתְרוּ דְּבֵקִים בַּמֶּלֶךְ. הַפִּלּוּג הָיָה גָּלוּי לָעַיִן.

*

בַּחֲצַר הַמֶּלֶךְ יָשַׁב דָּוִד עָיֵף, אַךְ נָחוּשׁ לְכַבּוֹת אֶת הָאֵשׁ מַהֵר, בְּטֶרֶם תִּתְפַּשֵּׁט. מוּלוֹ עָמַד עֲמָשָׂא, שַׂר הַצָּבָא הֶחָדָשׁ אֲשֶׁר מִנָּה תַּחַת יוֹאָב — אִישׁ תָּם, גְּלוּי פָּנִים. וַיְצַו עָלָיו דָּוִד פְּקֻדָּה בְּרוּרָה וּדְחוּפָה: "הַזְעֵק לִי אֶת אִישׁ יְהוּדָה שְׁלֹשֶׁת יָמִים, וְאַתָּה פֹּה עֲמֹד." (שמ"ב כ, ד)

יָצָא עֲמָשָׂא. וְהַזְּמַן נִמְתַּח. הִבִּיט דָּוִד אֶל הַיְצִיאָה — וַעֲמָשָׂא בּוֹשֵׁשׁ לָבוֹא. עָלָה הַלַּחַץ בְּלִבּוֹ, כִּי הַסַּכָּנָה בָּעִכּוּב, וּפָנָה אֶל אֲבִישַׁי: "עַתָּה יֵרַע לָנוּ שֶׁבַע בֶּן בִּכְרִי מִן אַבְשָׁלוֹם. אַתָּה קַח אֶת עַבְדֵי אֲדֹנֶיךָ וּרְדֹף אַחֲרָיו, פֶּן מָצָא לוֹ עָרִים בְּצֻרוֹת וְהִצִּיל עֵינֵנוּ." (שמ"ב כ, ו)

יָצָא יוֹאָב עִם הַגְּדוּדִים אֶל הַדֶּרֶךְ. וַעֲמָשָׂא בָּא לִקְרָאתָם — מְאַחֵר, אַךְ בּוֹטֵחַ, כְּאֶל אֶחָיו. וְיוֹאָב — קַר, חֶשְׁבּוֹן אֶחָד בְּעֵינָיו: הַשַּׂר אֲשֶׁר תָּפַס אֶת מְקוֹמוֹ נִצָּב לְפָנָיו.

"הֲשָׁלוֹם אַתָּה אָחִי?" (שמ"ב כ, ט) — קוֹלוֹ חַם, מְזֻיָּף, וְהַיָּד כְּבָר זוֹחֶלֶת אֶל הַלַּהַב. אָחַז בִּזְקַן עֲמָשָׂא בִּימִינוֹ, כְּנוֹשֵׁק לוֹ. וְעֲמָשָׂא הַתָּם הִתְקָרֵב, פָּנָיו נִפְתָּחוֹת אֵלָיו, וְלָחַשׁ בְּלֹא הֲבָנָה — אֱמוּנוֹ הוּא אָבְדָנוֹ: "אָחִי..."

וַיַּךְ יוֹאָב אֶל הַחֹמֶשׁ. אֵין מַכָּה שְׁנִיָּה. נָפַל עֲמָשָׂא, וְיוֹאָב פָּסַע מֵעָלָיו כְּמֵסִיר אֶבֶן מִן הַדֶּרֶךְ. הַדָּם עַל הַדֶּרֶךְ, וְהַגְּדוּדִים הָעוֹבְרִים נֶעֶצְרוּ רֶגַע, וְאַחַר נִמְשְׁכוּ אַחֲרֵי יוֹאָב — הַיָּד שֶׁאֵינָהּ נֶעֱצֶרֶת מוֹבִילָה שׁוּב. שֵׁנִית הַיָּד הַזֹּאת. כְּמוֹ אַבְנֵר בַּשַּׁעַר, כֵּן עֲמָשָׂא בַּדֶּרֶךְ. וְאֵין מִי שֶׁיַּעֲצֹר אֶת יַד יוֹאָב.

*

עַל אָבֵל בֵּית מַעֲכָה צָר יוֹאָב. הֵקִים סוֹלְלָה, וְהַחוֹמָה הֵחֵלָּה לְהִתְעַרְעֵר. וְאָז הוֹפִיעָה מֵעַל הַחוֹמָה אִשָּׁה — לֹא חֲמוּשָׁה. שְׁקֵטָה. סַמְכוּת שֶׁבָּאָה מִבִּפְנִים, לֹא מִן הַצָּבָא. לֹא הִתְחַנְּנָה — הִשִּׂיחָה.

"שִׁמְעוּ שִׁמְעוּ!" קָרְאָה, קוֹלָהּ נוֹשֵׂא מֶרְחָק. "אִמְרוּ נָא אֶל יוֹאָב: קְרַב עַד הֵנָּה וַאֲדַבְּרָה אֵלֶיךָ." (שמ"ב כ, טז)

הִתְקָרֵב יוֹאָב, וְרֶגַע שֶׁל דְּמָמָה נָפַל — שְׁנֵי קוֹלוֹת מְנֻגָּדִים יַעַמְדוּ זֶה מוּל זֶה: הַיָּד וְהַמִּלָּה. וְהִיא, בְּכֹבֶד רֹאשׁ, הִצִּיבָה אֶת הָעִיר וְאֶת עֶרְכָּהּ:

"דַּבֵּר יְדַבְּרוּ בָרִאשֹׁנָה לֵאמֹר: שָׁאֹל יִשְׁאֲלוּ בְּאָבֵל וְכֵן הֵתַמּוּ. אָנֹכִי שְׁלֻמֵי אֱמוּנֵי יִשְׂרָאֵל — אַתָּה מְבַקֵּשׁ לְהָמִית עִיר וְאֵם בְּיִשְׂרָאֵל. לָמָּה תְבַלַּע נַחֲלַת ה'?" (שמ"ב כ, יח-יט)

וְיוֹאָב — לָרִאשׁוֹנָה — נֶעֱצָר. לֹא בַּחֶרֶב נֶעֱצַר, כִּי אִם בַּמִּלָּה. תְּשׁוּבָתוֹ אַף הִיא קָרָה, אַךְ נְסוֹגָה, מוֹדֶה לֹא בַּלֵּב אֶלָּא בַּחֶשְׁבּוֹן: "חָלִילָה חָלִילָה לִי אִם אֲבַלַּע וְאִם אַשְׁחִית! לֹא כֵן הַדָּבָר, כִּי אִישׁ מֵהַר אֶפְרַיִם — שֶׁבַע בֶּן בִּכְרִי שְׁמוֹ — נָשָׂא יָדוֹ בַּמֶּלֶךְ בְּדָוִד. תְּנוּ אֹתוֹ לְבַדּוֹ, וְאֵלְכָה מֵעַל הָעִיר." (שמ"ב כ, כ-כא)

וְהָאִשָּׁה, שְׁקֵטָה וּבְטוּחָה, כְּמִי שֶׁהֶסְכֵּם נִסְגַּר בְּמִלָּה: "הִנֵּה רֹאשׁוֹ מֻשְׁלָךְ אֵלֶיךָ בְּעַד הַחוֹמָה." (שמ"ב כ, כא)

פָּנְתָה אֶל אַנְשֵׁי עִירָהּ, אֶל הַחֲשֵׁכָה אֲשֶׁר מֵאַחֲרֶיהָ. לֹא צָעֲקָה — שִׁכְנְעָה, וְחָכְמָתָהּ הָיְתָה נִשְׁקָהּ: "שִׁמְעוּ לִי, יוֹשְׁבֵי אָבֵל. נֶפֶשׁ אַחַת מְבַקֵּשׁ, וְעִיר שְׁלֵמָה יַחְמֹל. אִישׁ אֶחָד הֵבִיא עָלֵינוּ אֶת הַסֹּלְלָה — הֲנָמוּת כֻּלָּנוּ עַל גַּחַלְתּוֹ? לֹא בְחֶרֶב נִנָּצֵל, כִּי אִם בְּחָכְמָה."

אֲפֵלָה קְצָרָה. קוֹל תְּרוּעָה כְּבוּשָׁה מֵעֵבֶר לַחוֹמָה. וְאָז דְּבַר מָה נִזְרַק מֵעַל הַחוֹמָה, וְנָפַל לְרַגְלֵי יוֹאָב: רֹאשׁ שֶׁבַע. הִבִּיט בּוֹ יוֹאָב רֶגַע, תָּקַע בַּשּׁוֹפָר נְסִיגָה, וְהַגְּדוּדִים נָפוֹצוּ. הָעִיר נִצְּלָה. וְהָאִשָּׁה נוֹתְרָה לְבַדָּהּ עַל הַחוֹמָה, מַבִּיטָה אֶל הַצָּבָא הַנָּסוֹג. אִשָּׁה דִּבְּרָה — וְהַחֶרֶב שָׁבָה אֶל נְדָנָהּ.

*

מְאֻחָר בַּלַּיְלָה יָשַׁב דָּוִד לְבַדּוֹ, וּשְׁמוּעַת רֶצַח עֲמָשָׂא הִגִּיעָה אֵלָיו. וּבְלִבּוֹ עָלְתָה תְּפִלָּה אֶל מַעְלָה, לֹא אֶל אִישׁ:

> רִבּוֹנוֹ שֶׁל עוֹלָם,
> שְׁתֵּי יָדַיִם הִצִּילוּ לִי הַיּוֹם מַמְלָכָה —
> וְלֹא יָדַעְתִּי בְּאֵיזוֹ מֵהֶן לִבְחֹר.
>
> יַד יוֹאָב — קָרָה, מְהִירָה,
> אוֹחֶזֶת בַּזָּקָן וְדוֹקֶרֶת,
> אוֹמֶרֶת "אָחִי" וְשׁוֹפֶכֶת דָּם.
> בָּהּ נִכְבַּשׁ הַמֶּרֶד —
> וּבָהּ נִשְׁבַּר דְּבַר מָה שֶׁלֹּא יֵרָפֵא.
>
> וְיַד הָאִשָּׁה אֲשֶׁר עַל הַחוֹמָה —
> לֹא נָשְׂאָה חֶרֶב, לֹא שָׁפְכָה דָם,
> רַק דִּבְּרָה — וְעִיר שְׁלֵמָה נוֹתְרָה חַיָּה.
>
> אֱלֹהַי,
> לָמָּה הַהוֹרֵג עוֹמֵד לִימִינִי,
> וְהַחֲכָמָה — עַל חוֹמָה רְחוֹקָה,
> אֲשֶׁר לֹא אֵדַע אֶת שְׁמָהּ?
>
> לַמְּדֵנִי אֶת יָדָהּ.
> אֶת הַיָּד הָעוֹצֶרֶת,
> אֶת הַמִּלָּה הַמַּצֶּלֶת בִּמְקוֹם הַחֶרֶב —
> כִּי בָהּ, וְלֹא בְּדַם עֲמָשָׂא,
> תִּכּוֹן מַלְכוּת.

וְהַנֵּר דָּעַךְ, וְנִימַת כִּנּוֹר רִאשׁוֹנָה עָלְתָה בַּחֲשֵׁכָה.
