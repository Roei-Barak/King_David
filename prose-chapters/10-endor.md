# פֶּרֶק ט: בַּעֲלַת הָאוֹב בְּעֵין-דּוֹר

הַלַּיְלָה שֶׁלִּפְנֵי הַנְּפִילָה יָרַד עַל הַר הַגִּלְבֹּעַ כְּמַסָּךְ. רֵיחוֹת הַמִּלְחָמָה הַקְּרֵבָה נִדְּפוּ בָּאֲוִיר, וְהַצָּבָא רָבַץ סָבִיב — אֶלֶף עֵינַיִם פְּקוּחוֹת אֶל הַחֲשֵׁכָה. וְשָׁאוּל עָמַד לְבַדּוֹ. מֶלֶךְ יִשְׂרָאֵל, רֹאשׁוֹ מֵעַל לְכָל הָעָם מִשִּׁכְמוֹ וָמַעְלָה, נִצַּב כְּעֵץ שֶׁכָּרְתוּ אֶת שָׁרָשָׁיו וְעוֹדֶנּוּ זָקוּף. הַשָּׁמַיִם מֵעָלָיו שָׁתְקוּ.

הוּא דִּבֵּר אֶל הַחֲשֵׁכָה, בְּלַחַשׁ שֶׁאִישׁ לֹא נוֹעַד לְשָׁמְעוֹ. חָלַם — וְלֹא עָנָהוּ אִישׁ. שָׁאַל בְּנָבִיא — וְאֵין קוֹל. פָּנָה אֶל הָאוּרִים — וְשֶׁקֶט. וְאָז הֵרִים אֶת קוֹלוֹ אֶל הַמָּרוֹם, וְהַקּוֹל נִשְׁבַּר בְּעֶצֶם עֲלִיָּתוֹ: "אֵיפֹה אַתָּה?! מָה חָטָאתִי?!" הַשְּׁאֵלָה רִחֲפָה רֶגַע בָּאֲוִיר וְנָפְלָה אַרְצָה בְּלֹא מַעֲנֶה. שְׁתִיקָה מֻחְלֶטֶת עָנְתָה לוֹ, וְהַשְּׁתִיקָה הַזֹּאת הָיְתָה נוֹרָאָה מִכָּל תְּשׁוּבָה.

וְאָז, בְּשֶׁקֶט, פָּנָה אֶל עַבְדּוֹ: "בַּקְּשׁוּ לִי אֵשֶׁת בַּעֲלַת אוֹב — אֵלְכָה אֵלֶיהָ וְאֶדְרְשָׁה בָּהּ." הוּא בְּעַצְמוֹ הִכְרִית אֶת בַּעֲלֵי הָאוֹב מִן הָאָרֶץ, וְעַתָּה הָלַךְ אֶל מַה שֶּׁגֵּרַשׁ בְּיָדָיו. זֶה הָיָה הַהֵד הָאָפֵל לִשְׁתִיקַת הַשָּׁמַיִם — אַךְ בְּעוֹד יֵשׁ הַמִּתְחַזְּקִים מִתּוֹךְ הַשְּׁתִיקָה, שָׁאוּל יָרַד אֶל הַמֵּתִים.

וּמִתּוֹךְ הַחֲשֵׁכָה הַגְּמוּרָה בָּקְעָה מַנְגִּינַת מֵיתָר בּוֹדֵד, וְקוֹל — לֹא קוֹלוֹ שֶׁל שָׁאוּל, אֶלָּא כְּקוֹל הַגּוֹרָל עַצְמוֹ — נָשָׂא אֶת הַשִּׁיר:

> וּבְחֶשְׁכַת הַלַּיִל בְּלִי קֶשֶׁת וָשֶׁלַח
> עַל סוּס קַל עֵין-דּוֹרָה בָּא שָׁאוּל הַמֶּלֶךְ.
>
> וּבְאַחַד הַבָּתִּים אוֹר כֵּהֶה הוֹפִיעַ:
> –"פֹּה תָגוּר" – הַנַּעַר לוֹ חֶרֶשׂ הִבִּיעַ.
>
> –"אַתְּ בַּעֲלַת הָאוֹב? – "כֵּן, אֲדוֹנִי, הִנֵּנִי"
> –"נָא קָסְמִי בָּאוֹב, צֵל הָרֹאֶה הַרְאִינִי!"

הַשִּׁיר נָמוֹג, וְהַצְּלָלִית הִגִּיעָה אֶל הַדֶּלֶת. בַּיִת קָטָן בְּעֵין-דּוֹר, אוֹר כֵּהֶה בַּחַלּוֹן, וְשָׁאוּל מְחֻפָּשׂ עוֹמֵד מִחוּצָה לוֹ. הוּא דָּפַק, וְקוֹלוֹ נָמוּךְ: "קִסְמִי נָא לִי בְּאוֹב — וְהַעֲלִי לִי אֵת אֲשֶׁר אֹמַר אֵלָיִךְ." (שמ"א כח, ח)

הָאִשָּׁה נִרְתְּעָה, חַשְׁדָנִית. "הִנֵּה אַתָּה יָדַעְתָּ אֵת אֲשֶׁר עָשָׂה שָׁאוּל — אֲשֶׁר הִכְרִית אֶת הָאֹבוֹת וְאֶת הַיִּדְּעֹנִים מִן הָאָרֶץ! וְלָמָּה אַתָּה מִתְנַקֵּשׁ בְּנַפְשִׁי לַהֲמִיתֵנִי?!" בְּלֹא לָדַעַת, צָדְקָה: הָאִישׁ הָעוֹמֵד בְּשַׁעֲרָהּ הוּא שֶׁחָקַק אֶת הַחֹק שֶׁבִּקֵּשׁ אֶת רֹאשָׁהּ.

נִשְׁבַּע לָהּ שָׁאוּל: "חַי ה' — אִם יִקְּרָךְ עָוֹן בַּדָּבָר הַזֶּה." וּבַשְּׁבוּעָה הַזֹּאת הָיְתָה כָּל הַטְּרָגֶדְיָה — מֶלֶךְ שֶׁנִּשְׁבַּע בַּה' שֶׁשָּׁתַק לוֹ, כְּדֵי לַעֲבֹר עַל מִצְוַת ה' עַצְמוֹ.

"אֶת מִי אַעֲלֶה לָּךְ?" שָׁאֲלָה.

"אֶת שְׁמוּאֵל הַעֲלִי לִי."

הָאִשָּׁה הִתְקָרְבָה, הִבִּיטָה — וְצָרְחָה. הַצְּעָקָה קָרְעָה אֶת הַלַּיְלָה. "לָמָּה רִמִּיתָנִי?! וְאַתָּה שָׁאוּל!" בְּרֶגַע אֶחָד יָדְעָה אֶת מִי הֶעֱלְתָה, וְאֶת מִי שֵׁרְתָה.

הוֹרִיד שָׁאוּל אֶת הַמַּסְוֶה מֵעַל פָּנָיו. "אַל תִּירְאִי. מָה רָאִית?"

הִיא רָעֲדָה. "אֱלֹהִים רָאִיתִי עֹלִים מִן הָאָרֶץ... אִישׁ זָקֵן עֹלֶה — וְהוּא עֹטֶה מְעִיל." וְשָׁאוּל הִכִּיר אֶת הַמְּעִיל הַהוּא, אֶת הַמְּעִיל שֶׁכְּנָפוֹ נִקְרְעָה כְּבָר פַּעַם אַחַת בְּיָדָיו — וְהִשְׁתַּחֲוָה אַפַּיִם אָרְצָה.

וּמִן הַחֹשֶׁךְ — קוֹל, וְלֹא גּוּף: "לָמָּה הִרְגַּזְתַּנִי לְהַעֲלוֹת אֹתִי?" (שמ"א כח, טו)

וְשָׁאוּל עָנָה מִמַּעֲמַקִּים, וְכָל שִׁבְרוֹ עָלָה בַּמִּלִּים: "צַר לִי מְאֹד! וּפְלִשְׁתִּים נִלְחָמִים בִּי, וֵאלֹהִים סָר מֵעָלַי... וְלֹא עָנָנִי עוֹד — גַּם בְּיַד הַנְּבִיאִים — גַּם בַּחֲלֹמוֹת. דָּפַקְתִּי עַל כָּל הַשְּׁעָרִים — וְכֻלָּם נֶעֱלוּ. וָאֵרֵד הֲלוֹם — אֶל אֲשֶׁר גֵּרַשְׁתִּי בְיָדַי מִן הָאָרֶץ. זֹאת שְׁאֵרִית הַמֶּלֶךְ — לִקְרֹא אֶל הַמֵּתִים, כִּי הַחַיִּים שָׁתְקוּ. וָאֶקְרָאֲךָ לְהוֹדִיעֵנִי מָה אֶעֱשֶׂה."

שְׁתִיקָה. וְאָז דִּבֵּר שְׁמוּאֵל, בְּלֹא רַחֲמִים, כִּי כֵן הָאֱמֶת: "וְלָמָּה תִּשְׁאָלֵנִי? וַה' סָר מֵעָלֶיךָ וַיְהִי עָרֶךָ. כַּאֲשֶׁר שָׁמַעְתָּ בְּקוֹל ה' — וְלֹא עָשִׂיתָ... עַל כֵּן הַדָּבָר הַזֶּה עָשָׂה לְךָ ה' הַיּוֹם הַזֶּה. וְיִתֵּן ה' גַּם אֶת יִשְׂרָאֵל עִמְּךָ בְּיַד פְּלִשְׁתִּים — מָחָר אַתָּה וּבָנֶיךָ עִמִּי." (שמ"א כח, יט)

שְׁתִיקָה מֻחְלֶטֶת. וְאָז נָפַל שָׁאוּל כְּאֶרֶץ, מְלוֹא קוֹמָתוֹ. בְּלִי צְעָקָה. פָּשׁוּט — נָפַל.

הָאִשָּׁה נִגְּשָׁה, הִבִּיטָה בּוֹ. "הִנֵּה שָׁמְעָה שִׁפְחָתְךָ בְּקֹלֶךָ... וְעַתָּה — שְׁמַע נָא גַם אַתָּה בְּקוֹל שִׁפְחָתֶךָ... שִׂימִי לְפָנֶיךָ פַּת לֶחֶם — וֶאֱכֹל..." בַּתְּחִלָּה סֵרֵב, גּוּפוֹ מְעֻוָּת בִּמְאוּן. וְאָז נִצַּח הַגּוּף אֶת הַנֶּפֶשׁ, וְהוּא אָכַל — לְאַט, בְּלִי טַעַם, כְּאָדָם שֶׁכְּבָר אֵינֶנּוּ שָׁם.

וְהָאִשָּׁה הִבִּיטָה בּוֹ: מֶלֶךְ גָּדוֹל, שָׁבוּר, אוֹכֵל עַל הָרִצְפָּה — בַּלַּיְלָה, בְּבֵית הָאוֹב שֶׁגֵּרַשׁ, מִן הַיָּד שֶׁאָסַר. כָּל מַסַּע נְפִילָתוֹ הִצְטַמְצֵם לִתְמוּנָה אַחַת. וְאָז קָם. רֶגַע אֶחָד הִזְדַּקֵּף לִמְלוֹא קוֹמָתוֹ, כְּמוֹ שֶׁהָיָה — וְאָז נִשְׁמְטָה הַקּוֹמָה, וְהוּא יָצָא אֶל הַחֹשֶׁךְ.

וְשָׁאוּל נִבְלַע בַּחֲשֵׁכָה, וְאוֹתוֹ קוֹל חָזַר — אוֹתוֹ לַחַן, שָׁקֵט יוֹתֵר, שָׁבוּר יוֹתֵר. עַתָּה כְּבָר יָדַע הַשּׁוֹמֵעַ אֶת אֲשֶׁר הַשִּׁיר יָדַע מֵרֹאשׁ:

> –עַל מֶרְיְךָ, גְּאוֹן לִבְּךָ אֱלֹהִים יִזְעָמְךָ!
> מָחָר אַתָּה עִמִּי, גַּם אַתָּה, גַּם עַמְּךָ!
>
> בְּאַשְׁמֹרֶת הַבֹּקֶר בְּלִי קֶשֶׁת וָשֶׁלַח
> עַל סוּס קַל הַמַּחֲנֶה שָׁב שָׁאוּל הַמֶּלֶךְ,
>
> וּפָנָיו חָוָרוּ, אַךְ בְּלִבּוֹ אֵין מוֹרָא,
> וּבְעֵינָיו מִתְנוֹצְצוֹת – הַיֵּאוּשׁ הַנּוֹרָא.

הַלַּחַן נֶחְתַּם. חֹשֶׁךְ מֻחְלָט. מִי שֶׁיָּצָא — שָׁב; אַךְ לֹא אוֹתוֹ אָדָם. וּמָחָר — גִּלְבֹּעַ.
