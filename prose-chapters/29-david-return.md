# פֶּרֶק כח: שִׁיבַת דָּוִד

מַחֲנֵה מַחֲנַיִם שָׁכַן בְּדוּמִיָּה כְּמִי שֶׁנִּצַּח וְלֹא יָדַע מַה לַּעֲשׂוֹת בְּנִצְחוֹנוֹ. הָאֲנָשִׁים שֶׁשָּׁבוּ מִן הַשָּׂדֶה, חַרְבּוֹתֵיהֶם עוֹד אֲדֻמּוֹת, הִתְגַּנְּבוּ אֶל הָעִיר כְּמִתְבַּיְּשִׁים — כִּי הַמֶּלֶךְ יָשַׁב בָּעֲלִיָּה וּבָכָה. קוֹלוֹ עָבַר אֶת הַקִּירוֹת וְנִשְׁבַּר עַל לֵב כָּל אִישׁ: "בְּנִי אַבְשָׁלוֹם, בְּנִי בְנִי אַבְשָׁלוֹם." הַשִּׂמְחָה שֶׁל הַיּוֹם הַזֶּה נֶהְפְּכָה לְאֵבֶל, וְהָעָם נִגְנַב אֶל הָעִיר כְּגַנֹּב הָעָם הַנִּכְלָמִים בְּנוּסָם בַּמִּלְחָמָה.

וַיֵּדַע יוֹאָב בֶּן צְרוּיָה כִּי אֵין עוֹד עֵת לְרַחֵם עַל הַמֶּלֶךְ הַבּוֹכֶה. נִכְנַס אֶל הָעֲלִיָּה וְעָמַד בַּפֶּתַח, וְהִבִּיט בּוֹ אֲרֻכּוֹת — אִישׁ הַמִּלְחָמָה שֶׁרָאָה כְּבָר אֶת מְחִיר הַחֻלְשָׁה — וְאָז דִּבֵּר, יָשָׁר, בְּלֹא רַכּוּת:

"בּוֹשׁוֹת הַיּוֹם אֶת פְּנֵי כָל עֲבָדֶיךָ — הַמְּמַלְּטִים אֶת נַפְשְׁךָ הַיּוֹם... כִּי אָהַבְתָּ אֶת שֹׂנְאֶיךָ וְשָׂנֵאתָ אֶת אֹהֲבֶיךָ! כִּי הוֹדַעְתָּ הַיּוֹם כִּי אֵין לְךָ שָׂרִים וַעֲבָדִים. כִּי יָדַעְתִּי הַיּוֹם כִּי לוּ אַבְשָׁלוֹם חַי וְכֻלָּנוּ הַיּוֹם מֵתִים — אָז יָשָׁר בְּעֵינֶיךָ! עַתָּה קוּם, צֵא וְדַבֵּר עַל לֵב עֲבָדֶיךָ."

דָּוִד נָשָׂא אֵלָיו עֵינַיִם אֲדֻמּוֹת. אָרֹךְ הִבִּיט בְּשַׂר צְבָאוֹ, בָּאִישׁ שֶׁהֵמִית אֶת בְּנוֹ וְהֵצִיל אֶת מַלְכוּתוֹ בְּמַעֲשֶׂה אֶחָד. וְאָז קָם. כִּי יָדַע שֶׁהַצֶּדֶק עִם הָאַכְזָר, וְשֶׁמֶּלֶךְ אֵינוֹ רַשַּׁאי לִבְכּוֹת לָנֶצַח.

*

הַיַּרְדֵּן הִשְׁתָּרַע לְפָנָיו כִּגְבוּל בֵּין מַה שֶּׁהָיָה לְמַה שֶּׁיִּהְיֶה. עוֹד טֶרֶם נָגְעוּ רַגְלָיו בַּמַּיִם, מִהֵר אִישׁ לִקְרָאתוֹ, רָץ לְפָנָיו, וְעִמּוֹ אֶלֶף אִישׁ מִבִּנְיָמִין. שִׁמְעִי בֶּן גֵּרָא — הוּא הָאִישׁ אֲשֶׁר קִלֵּל וְזָרַק אֲבָנִים — נָפַל אַרְצָה לִפְנֵי הַמֶּלֶךְ בְּטֶרֶם עָבְרוּ אֶת הַמַּיִם.

וּבְעָבְרוֹ עָלְתָה בְּאָזְנֵי דָּוִד הַקְּלָלָה הַיְשָׁנָה, כְּאִלּוּ נֶחְרְתָה בְּקוֹלוֹ שֶׁל הָאִישׁ הַזֶּה לְעוֹלָם: "צֵא צֵא אִישׁ הַדָּמִים וְאִישׁ הַבְּלִיָּעַל." הָאֲבָנִים שֶׁעָפוּ, הֶעָפָר שֶׁזָּרַק לְעֻמָּתוֹ בְּיוֹם בְּרִיחָתוֹ הַשָּׁפֵל. וְעַתָּה אוֹתוֹ הַפֶּה — בֶּעָפָר.

"אַל יַחֲשָׁב לִי אֲדֹנִי עָוֹן," מִהֵר שִׁמְעִי, פָּנָיו בָּאָרֶץ וְקוֹלוֹ נִשְׁבָּר מֵאֵימָה, נֶחְפָּז פֶּן יַקְדִּימֵהוּ הַמֶּלֶךְ בְּדָבָר. "וְאַל תִּזְכֹּר אֵת אֲשֶׁר הֶעֱוָה עַבְדְּךָ בַּיּוֹם אֲשֶׁר יָצָא אֲדֹנִי הַמֶּלֶךְ מִירוּשָׁלָיִם — אַל יָשֵׂם הַמֶּלֶךְ אֶל לִבּוֹ. כִּי יָדַע עַבְדְּךָ כִּי אֲנִי חָטָאתִי, וְהִנֵּה בָאתִי הַיּוֹם רִאשׁוֹן לְכָל בֵּית יוֹסֵף לָרֶדֶת לִקְרַאת אֲדֹנִי הַמֶּלֶךְ."

הִתְרַפֵּס וְנָשַׁק אֶת הֶעָפָר אֲשֶׁר זָרַק בּוֹ אֶתְמוֹל. רָץ רִאשׁוֹן מִכֻּלָּם — לֹא מֵאַהֲבָה, אֶלָּא מִפַּחַד. יָדַע הָאִישׁ: הַיּוֹם שָׁב דָּוִד מֶלֶךְ, וְהַמְּקַלֵּל אִישׁ הַדָּמִים יִחְיֶה רַק אִם יַקְדִּים וְיִתְחַנֵּן.

אֲבִישַׁי בֶּן צְרוּיָה לָהַט. יָדוֹ נָחָה כְּבָר עַל נְצַב הַחֶרֶב, וְהוּא רָאָה לְפָנָיו לֹא מִתְחַנֵּן אֶלָּא אוֹיֵב מֻבָּס. הַקִּנְאָה בָּעֲרָה בּוֹ: לִמְשִׁיחַ ה' לֹא יְקַלֵּל אִישׁ וְיִחְיֶה. וַיִּלְחַשׁ אֶל דָּוִד, חַד אַךְ נִמְרָץ: "הֲתַחַת זֹאת לֹא יוּמַת שִׁמְעִי, כִּי קִלֵּל אֶת מְשִׁיחַ ה'?"

דָּוִד הִבִּיט בַּמִּתְרַפֵּס שֶׁבֶּעָפָר. אֲרֻכּוֹת הִבִּיט. יָדוֹ לֹא נָעָה אֶל הַחֶרֶב וְלֹא צִוְּתָה. וְאָז פָּנָה אֶל אֲבִישַׁי, בְּשֶׁקֶט וּבַעֲיֵפוּת שֶׁל אִישׁ שֶׁרָאָה כְּבָר יוֹתֵר מִדַּי דָּם:

"מַה לִּי וְלָכֶם בְּנֵי צְרוּיָה, כִּי תִהְיוּ לִי הַיּוֹם לְשָׂטָן? הַיּוֹם יוּמַת אִישׁ בְּיִשְׂרָאֵל? כִּי הֲלוֹא יָדַעְתִּי כִּי הַיּוֹם אֲנִי מֶלֶךְ עַל יִשְׂרָאֵל." (שמ"ב יט, כג)

פָּתַח אֲבִישַׁי פֶּה לְהָשִׁיב — אַךְ דָּוִד הֵרִים יָד וְלֹא נָתַן. דַּי. וַיֵּדַע דָּוִד הֵיטֵב שֶׁהָאִישׁ מְשַׁקֵּר, יָדַע שֶׁבַּיּוֹם שֶׁיִּרְפֶּה כֹּחוֹ יְקַלֵּל שׁוּב, יָדַע שֶׁאֵין כָּאן חֲרָטָה אֶלָּא חֶשְׁבּוֹן שֶׁל פַּחַד. וּבְכָל זֹאת — לֹא הַיּוֹם. יָדוֹ נֶעֶצְרָה. כְּמוֹ בִּמְעָרַת עֵין גֶּדִי, כְּשֶׁחָתַךְ כְּנַף מְעִיל שָׁאוּל וְלֹא נָטַל נֶפֶשׁ — אָז כְּלַפֵּי רוֹדְפוֹ הַמֶּלֶךְ, וְכָאן כְּלַפֵּי מְקַלְלוֹ. כֹּחַ בַּיָּד, וְרִסּוּן בַּלֵּב.

קְרָאוּ עֵינָיו אֶת פְּנֵי הָאִישׁ, וְאַחַר שְׁתִיקָה נִשְׁבַּע לוֹ, קוֹלוֹ שָׁקֵט אַךְ חָתוּם: "לֹא תָמוּת." (שמ"ב יט, כד)

הִבִּיט בּוֹ שִׁמְעִי כְּלֹא מַאֲמִין, וְנָשַׁם. הַחַיִּים שֶׁנִּתְּנוּ לוֹ בְּרֶגַע זֶה הָיוּ מַתָּנָה שֶׁאֵינוֹ רָאוּי לָהּ, וּשְׁנֵיהֶם יָדְעוּ זֹאת. וּבְעוֹד דָּוִד פּוֹנֶה לָלֶכֶת, לָחַשׁ רַק לְעַצְמוֹ: "הַיּוֹם אֲנִי מֶלֶךְ — וּמֶלֶךְ אֵינוֹ הוֹרֵג בְּיוֹם שִׂמְחָתוֹ. לֹא קֶטֶל מְקַלֵּל אֲבַקֵּשׁ, כִּי אִם לֵב נִשְׁבָּר וְנִדְכֶּה. וּמָחָר — יִשְׁפֹּט אֱלֹהִים."

בַּצַּד חָרַק אֲבִישַׁי שִׁנָּיו, וְהַחֶרֶב שָׁבָה אֶל נְדָנָהּ בְּחֹסֶר רָצוֹן. שִׁמְעִי הֵרִים רֹאשׁ וְהִבִּיט אַחֲרֵי דָּוִד — וְאֵין לָדַעַת אִם הַכְנָעָה הָיְתָה זוֹ אוֹ חֶשְׁבּוֹן.

*

מְפִיבֹשֶׁת יָרַד לִקְרַאת הַמֶּלֶךְ, וְעָלָיו אוֹתוֹת הָאֵבֶל הָאֱמֶת: לֹא רָחַץ רַגְלָיו, לֹא עָשָׂה שְׂפָמוֹ, וְאֶת בְּגָדָיו לֹא כִבֵּס לְמִן הַיּוֹם אֲשֶׁר יָצָא הַמֶּלֶךְ עַד הַיּוֹם אֲשֶׁר שָׁב בְּשָׁלוֹם.

"לָמָּה לֹא הָלַכְתָּ עִמִּי, מְפִיבֹשֶׁת?" שָׁאַל דָּוִד.

"אֲדֹנִי הַמֶּלֶךְ, עַבְדִּי רִמָּנִי," הֵשִׁיב בֶּן יְהוֹנָתָן. "כִּי אָמַר עַבְדְּךָ: אֶחְבְּשָׁה לִי הַחֲמוֹר וְאֶרְכַּב עָלֶיהָ וְאֵלֵךְ אֶת הַמֶּלֶךְ — כִּי פִסֵּחַ עַבְדֶּךָ. וַיְרַגֵּל בְּעַבְדְּךָ אֶל אֲדֹנִי הַמֶּלֶךְ."

צִיבָא עָמַד שָׁם, וְאֵין לָדַעַת מִי מִשְּׁנֵיהֶם דּוֹבֵר אֱמֶת. הִבִּיט דָּוִד בֵּין הַשְּׁנַיִם, וְלֹא הָיָה בּוֹ עוֹד כֹּחַ לְדִין וּלְמִשְׁפָּט. אַחַר שְׁתִיקָה אָמַר, בְּקֹצֶר רוּחַ שֶׁל עֲיֵפוּת, לֹא שֶׁל כַּעַס: "לָמָּה תְּדַבֵּר עוֹד דְּבָרֶיךָ? דִּבַּרְתִּי — אַתָּה וְצִיבָא תַּחְלְקוּ אֶת הַשָּׂדֶה."

וְאָז קָרָה דָּבָר שֶׁלֹּא צִפָּה לוֹ דָּוִד. עָנָה מְפִיבֹשֶׁת בְּנַחַת, בְּלֹא טִינָה: "גַּם אֶת הַכֹּל יִקָּח, אַחֲרֵי אֲשֶׁר בָּא אֲדֹנִי הַמֶּלֶךְ בְּשָׁלוֹם אֶל בֵּיתוֹ. הַשָּׂדֶה — שֶׁל צִיבָא. הַשֵּׁם — שֶׁל יְהוֹנָתָן. דַּי לִי שֶׁשַּׁבְתָּ."

הִבִּיט בּוֹ דָּוִד. זֶה הָאִישׁ שֶׁיָּשַׁב בְּאֵבֶל וְלֹא רָחַץ רַגְלָיו כָּל יְמֵי הַבְּרִיחָה, וְאֵינֶנּוּ מְבַקֵּשׁ דָּבָר זוּלָתִי פְּנֵי הַמֶּלֶךְ. וַיֵּדַע דָּוִד: כָּךְ אָהַב אָבִיו. כָּךְ בְּדִיּוּק.

*

וּבַרְזִלַּי הַגִּלְעָדִי בָּא — זָקֵן מְאֹד, בֶּן שְׁמֹנִים שָׁנָה, אֲשֶׁר כִּלְכֵּל אֶת הַמֶּלֶךְ בְּמָזוֹן בִּימֵי בְרִיחָתוֹ. וְדָוִד, בְּחֹם אֲמִתִּי, אָמַר לוֹ: "עֲבֹר אִתִּי, וְכִלְכַּלְתִּי אֹתְךָ עִמָּדִי בִּירוּשָׁלָיִם."

חִיֵּךְ בַּרְזִלַּי חִיּוּךְ עָצוּב: "כַּמָּה יְמֵי שְׁנֵי חַיַּי, כִּי אֶעֱלֶה אֶת הַמֶּלֶךְ יְרוּשָׁלָיִם? בֶּן שְׁמֹנִים שָׁנָה אָנֹכִי הַיּוֹם — הַאֵדַע בֵּין טוֹב לְרָע? הֲיִטְעַם עַבְדְּךָ אֶת אֲשֶׁר אֹכַל וְאֶת אֲשֶׁר אֶשְׁתֶּה? הַאֶשְׁמַע עוֹד בְּקוֹל שָׁרִים וְשָׁרוֹת? וְלָמָּה יִהְיֶה עַבְדְּךָ עוֹד לְמַשָּׂא אֶל אֲדֹנִי הַמֶּלֶךְ? יַעֲבֹר נָא עַבְדְּךָ — וְאָמֻת בְּעִירִי."

הֶחֱזִיק דָּוִד אֶת יָדוֹ, וּשְׁנֵיהֶם בָּכוּ. לֹא עַל הַפְּרֵדָה הַזֹּאת לְבַדָּהּ בָּכוּ, כִּי אִם עַל כָּל מִי שֶׁכְּבָר אֵינֶנּוּ: יוֹנָתָן, אַבְשָׁלוֹם, הַשָּׁנִים שֶׁחָלְפוּ. כִּי בַּרְזִלַּי הָיָה אַחֲרוֹן הַזְּקֵנִים שֶׁזּוֹכְרִים אֶת דָּוִד הַנַּעַר. הַנְּשִׁיקָה אֲשֶׁר נָשְׁקוּ זֶה לָזֶה הָיְתָה רֶגַע שֶׁל חַיִּים בְּתוֹךְ כָּל הַמָּוֶת — כִּי אַחֲרֵי כָל הָאֵבֶל, עוֹד נִמְצָא אָדָם אֶחָד שֶׁאוֹהֵב. וַיָּשָׁב בַּרְזִלַּי לִמְקוֹמוֹ.

הִבִּיט דָּוִד אַחֲרָיו וְעָצַם עֵינָיו. עוֹד אֶחָד נִפְרָד. וְלָחַשׁ, לֹא לְאִישׁ: "הַשְׁלֵךְ עַל ה' יְהָבְךָ וְהוּא יְכַלְכְּלֶךָ." (תהילים נה, כג)

*

בַּגִּלְגָּל עָבַר דָּוִד אֶת הַיַּרְדֵּן, וְכָל עַם יְהוּדָה הֶעֱבִירוּ אֶת הַמֶּלֶךְ, וְעִמָּם חֲצִי עַם יִשְׂרָאֵל. וְאָז, בְּעֶצֶם רֶגַע הַשִּׁיבָה, נִשְׁמַע קוֹל מִתְגַּלֵּעַ — לֹא אוֹיֵב מִבַּחוּץ, כִּי אִם קֶרַע מִבִּפְנִים. נִגְּשׁוּ אַנְשֵׁי יִשְׂרָאֵל אֶל הַמֶּלֶךְ, וּפְנֵיהֶם בּוֹעֲרוֹת מֵעֶלְבּוֹן הַמֻּדָּר:

"מַדּוּעַ גְּנָבוּךָ אַחֵינוּ אִישׁ יְהוּדָה, וַיַּעֲבִרוּ אֶת הַמֶּלֶךְ וְאֶת בֵּיתוֹ אֶת הַיַּרְדֵּן?" (שמ"ב יט, מב)

וְאַנְשֵׁי יְהוּדָה הֵשִׁיבוּ בִּגְבוּרָה, בְּקִרְבָה שֶׁל בְּעָלִים — הַמֶּלֶךְ שֶׁלָּהֶם, מִבְּשָׂרָם: "כִּי קָרוֹב הַמֶּלֶךְ אֵלַי. וְלָמָּה זֶּה חָרָה לְךָ עַל הַדָּבָר הַזֶּה? הֶאָכֹל אָכַלְנוּ מִן הַמֶּלֶךְ, אִם נִשֵּׂאת נִשָּׂא לָנוּ?" (שמ"ב יט, מג)

עָלָה קוֹל אַנְשֵׁי יִשְׂרָאֵל — לֹא עוֹד עֶלְבּוֹן בִּלְבַד, כִּי אִם תְּבִיעַת בְּכוֹרָה: "עֶשֶׂר יָדוֹת לִי בַמֶּלֶךְ, וְגַם בְּדָוִד אֲנִי מִמְּךָ. וּמַדּוּעַ הֱקִלֹּתַנִי, וְלֹא הָיָה דְבָרִי רִאשׁוֹן לִי לְהָשִׁיב אֶת מַלְכִּי?" (שמ"ב יט, מד)

וְאַנְשֵׁי יְהוּדָה הֵשִׁיבוּ קָשׁוֹת. הַקּוֹלוֹת הִתְעַרְבְּבוּ, הִתְגַּבְּרוּ, נֶחְתְּכוּ זֶה בָזֶה — שְׁנֵי גּוּשֵׁי קוֹל הַצּוֹעֲקִים מֵעַל רֹאשׁוֹ שֶׁל הַמֶּלֶךְ. דָּוִד עָמַד בָּאֶמְצַע, בֵּין הַשְּׁבָטִים, וְלֹא הִצְלִיחַ לְהַשְׁתִּיק. הַנִּצָּחוֹן עַל אַבְשָׁלוֹם עוֹד טְרִי, וְהִנֵּה כְּבָר נִפְעַר קֶרַע חָדָשׁ. שָׁב מֶלֶךְ — אַךְ הַמַּמְלָכָה אֲשֶׁר אֵלֶיהָ שָׁב כְּבָר מִתְבַּקַּעַת.

וַיֶּחֱזַק דְּבַר אִישׁ יְהוּדָה מִדְּבַר אִישׁ יִשְׂרָאֵל. (שמ"ב יט, מד)

הִבִּיט דָּוִד מִצַּד אֶל צַד, עָיֵף. רָאָה הַיּוֹם אֶת שִׁמְעִי מְקַלֵּל וּמִתְחַנֵּן, אֶת מְפִיבֹשֶׁת מְוַתֵּר, אֶת בַּרְזִלַּי נִפְרָד — וְעַתָּה, בְּרֶגַע הַשִּׁיבָה עַצְמוֹ, רָאָה אֶת הָאַחְדוּת נִקְרַעַת שׁוּב מִתַּחַת יָדָיו. לֹא בַּחֶרֶב הַפַּעַם, כִּי אִם בְּמִלִּים. וַיֵּדַע: כָּךְ מַתְחִיל הַכֹּל.

וּבְלִבּוֹ עָלְתָה תְּפִלָּה, לֹא לְאִישׁ:

> שַׁבְתִּי — וְהָעָם בְּקוֹלוֹ נֶחְלָק.
> לֹא חֶרֶב אַבְשָׁלוֹם הַיּוֹם,
> כִּי אִם לְשׁוֹן אָח בְּאָחִיו.
> מִי יִתֵּן וְיֵשְׁבוּ אַחִים גַּם יָחַד —
> וַאֲנִי, בְּתוֹכָם, וְאֵין בְּיָדִי לְאַחוֹת אֶת הַשֶּׁבֶר.

הַקּוֹלוֹת עוֹד הִתְגַּבְּרוּ. וְאָז, מִתּוֹךְ הַהֲמֻלָּה, גָּבַר קוֹל אֶחָד עַל כֻּלָּם — חַד כְּשׁוֹפָר. אִישׁ בְּלִיָּעַל. וּשְׁמוֹ שֶׁבַע בֶּן בִּכְרִי.
