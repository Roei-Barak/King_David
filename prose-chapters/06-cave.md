# פֶּרֶק ה: הַמְּעָרָה — לֵדַת הַכִּנּוֹר

בְּמַעֲמַקֵּי מְעָרַת עֵין גֶּדִי שָׁכַן חֹשֶׁךְ עָמֹק, וְטִפְטוּף מַיִם אִטִּי נָקַף בּוֹ כְּשָׁעוֹן שֶׁאֵין לוֹ סוֹף. הַרְחֵק, אֶל מוּל הַפֶּתַח, סִנְוֵר אוֹר הַיּוֹם — אַךְ הוּא לֹא הִגִּיעַ עַד אֶל אַנְשֵׁי דָוִד, שֶׁהָיוּ פְּזוּרִים בַּצְּלָלִים, מֻתָּשִׁים. אֶחָד לִטֵּשׁ פִּגְיוֹן בְּלִי דַּעַת, אַחֵר עִסָּה רֶגֶל פְּצוּעָה, וְדָוִד יָשַׁב בְּנִפְרָד מִכֻּלָּם, עֵינָיו נְעוּצוֹת בַּפֶּתַח הָרָחוֹק.

אֲבִישַׁי שָׁבַר אֶת הַדְּמָמָה, וְקוֹלוֹ שָׁקֵט וּמָרִיר. שָׁלֹשׁ שָׁנִים, אָמַר, נָע וָנָד מִמְּעָרָה לִמְעָרָה, מִסֶּלַע אֶל סֶלַע; טַעַם לֶחֶם בֵּיתָם נֶעֱלַם מִפִּיהֶם, וְלֶחֶם הַדְּאָגָה אָכְלוּ יָמִים רַבִּים. אֶחָד מִן הָאֲנָשִׁים הֵרִים אֵלָיו אֶת קוֹלוֹ הַשָּׁבוּר: מְשָׁחֲךָ שְׁמוּאֵל לְמֶלֶךְ, אָמַר, וְאָנוּ נֶחְבָּאִים בַּחֲשֵׁכָה כְּצִפּוֹר בּוֹדֵד עַל גָּג. מָתַי, דָּוִד? מָתַי יָקוּם הַדָּבָר?

דָּוִד לֹא הֵסֵב אֶת מַבָּטוֹ מִן הַפֶּתַח. רַךְ הָיָה קוֹלוֹ, אַךְ יַצִּיב כְּסֶלַע. לֹא בְּיָדִי הַיּוֹם וְהַשָּׁעָה, הֵשִׁיב. בְּיָדִי רַק דָּבָר אֶחָד — לֹא לְקַלְקֵל אֶת הַדֶּרֶךְ. גּוֹל עַל ה' דַּרְכֶּךָ, וּבְטַח עָלָיו וְהוּא יַעֲשֶׂה. וְאַחַר הוֹסִיף, כְּמִי שֶׁמְּדַבֵּר אֶל לִבּוֹ יוֹתֵר מֵאֶל אֲנָשָׁיו: לֹא אֶשָּׂא נֵזֶר בְּדָם, פֶּן יִדְבַּק הַדָּם בְּיָדַי כָּל יָמָי.

אֲבִישַׁי פָּתַח פִּיו לְהָשִׁיב — וְקָפָא. צֵל מִלֵּא אֶת פֶּתַח הַמְּעָרָה.

דְּמוּת נִכְנְסָה אֶל תּוֹךְ הָאוֹר: מֶלֶךְ, לְבַדּוֹ, בְּלִי שׁוֹמְרִים. שָׁאוּל. הוּא פָּנָה הַצִּדָּה, הִנִּיחַ אֶת חֲנִיתוֹ וְאֶת מְעִילוֹ, וְהִתְיַשֵּׁב — חָשׂוּף לְגַמְרֵי, גַּבּוֹ אֶל מַעֲמַקֵּי הַמְּעָרָה, וְלֹא יָדַע כִּי עַשְׂרוֹת עֵינַיִם נְעוּצוֹת בּוֹ מִן הַחֹשֶׁךְ.

הוּא, לָחַשׁ אֲבִישַׁי בְּקוֹל שֶׁכִּמְעַט לֹא הָיָה לוֹ קוֹל. שָׁאוּל. לְבַדּוֹ. בְּלִי חֲנִית.

שֶׁקֶט מָתוּחַ עָמַד בָּאֲוִיר, וְאָז עָלָה הַמִּלְמוּל — לְחִישָׁה שֶׁל אֲנָשִׁים מִתְגַּנֶּבֶת כְּפִתּוּי הַמְטַפֵּס. הִנֵּה הַיּוֹם, לָחֲשׁוּ, הִנֵּה הַיּוֹם אֲשֶׁר אָמַר ה': הִנֵּה אָנֹכִי נֹתֵן אֶת אֹיִבְךָ בְּיָדֶךָ, וְעָשִׂיתָ לּוֹ כַּאֲשֶׁר יִטַב בְּעֵינֶיךָ. וַאֲבִישַׁי, נוֹאָשׁ וְנִלְהָב, הִפְצִיר: מִלָּה אַחַת, דָּוִד. רַק תֵּן אוֹת, וְתֵם הַמִּרְדָּף, תֵּם הַמְּעָרוֹת, תַּמּוּ הַלֵּילוֹת בְּלִי שֵׁם. הַלַּיְלָה תִּישַׁן בְּמִטַּת מֶלֶךְ. אֱלֹהִים בְּעַצְמוֹ פָּתַח אֶת הַשַּׁעַר!

דָּוִד הִבִּיט בְּשָׁאוּל הַפּוֹנֶה לוֹ עֹרֶף, וּפָנָיו נִקְרְעוּ. שָׁלַף סַכִּין, סִמֵּן לַאֲנָשָׁיו לְהִשָּׁאֵר בִּמְקוֹמָם, וְהִתְגַּנֵּב קָדִימָה צַעַד אַחַר צַעַד, בְּלִי קוֹל, עַד שֶׁעָמַד מֵאֲחוֹרֵי הַמֶּלֶךְ. הֵרִים אֶת יָדוֹ —

וְהַיָּד עָצְרָה בָּאֲוִיר, מֵעַל עָרְפּוֹ הֶחָשׂוּף שֶׁל שָׁאוּל. רָעֲדָה. לֹא הָיָה זֶה פַּחַד, כִּי אִם הַיֵּצֶר עַצְמוֹ. הַלַּהַב כֹּה קָרוֹב הָיָה, וְשָׁלֹשׁ שְׁנוֹת מְעָרוֹת וְרָעָב וְלֵילוֹת בְּלִי שֵׁם דָּחֲפוּ כֻּלָּן אֶת הַיָּד לְמַטָּה. דָּוִד רָצָה. רֶגַע אָרֹךְ, בִּלְתִּי נִסְבָּל, נִתְלְתָה הַיָּד בֵּין בָּשָׂר לֶאֱמוּנָה. וּבְלַחַשׁ שֶׁכִּמְעַט לֹא נִשְׁמַע, אֶל עַצְמוֹ וְלֹא אֶל שָׁאוּל, אָמַר: יָדִי שֶׁלִּי — וְאֵינֶנָּה שֶׁלִּי.

וּבָרֶגַע הָאַחֲרוֹן, הַיָּד הָרוֹעֶדֶת לֹא צָנְחָה אֶל הַבָּשָׂר, אֶלָּא נֶחְתְּכָה הַצִּדָּה: בִּמְקוֹם לִדְקֹר, כָּרְתָה חֶרֶשׁ אֶת כְּנַף הַמְּעִיל. הָאַלִּימוּת הוּסְבָה מִן הַגּוּף אֶל הַסֵּמֶל. דָּוִד נָסוֹג אֶל הַצְּלָלִים, פִּסַּת בַּד רוֹעֶדֶת בְּיָדוֹ, הִבִּיט בָּהּ — וּפָנָיו הִשְׁתַּנּוּ לְגַמְרֵי. חָנוּק, לְעַצְמוֹ, מִלְמֵל: אַךְ כְּנַף בֶּגֶד, וְלֹא יוֹתֵר מִזֶּה לָקַחְתִּי. וְלִבּוֹ הִכָּה אוֹתוֹ. וְגַם זֹאת, אָמַר, כָּבְדָה עָלַי לָשֵׂאת.

אֲבִישַׁי לָחַשׁ בְּזַעַם, יָדוֹ מוּשֶׁטֶת: מָה זֶה עָשִׂיתָ? חֲצִי הַדֶּרֶךְ עָשִׂיתָ — וְשָׁבְתָּ? תְּנֵהוּ לִי, וַאֲנִי אֲכַלֶּנּוּ; פַּעַם אַחַת, וְלֹא אֶשְׁנֶה לוֹ.

דָּוִד הִסְתּוֹבֵב אֶל אֲנָשָׁיו, וְקוֹלוֹ נָמוּךְ וְחוֹתֵךְ עָצַר אוֹתָם כְּמוֹ קִיר. חָלִילָה לִּי מֵה', אָמַר; חָלִילָה לִּי מֵה' אִם אֶעֱשֶׂה אֶת הַדָּבָר הַזֶּה לַאדֹנִי, לִמְשִׁיחַ ה', לִשְׁלֹחַ יָדִי בּוֹ — כִּי מְשִׁיחַ ה' הוּא. אֲבִישַׁי נָשַׁם בִּכְבֵדוּת, וְהָאֲנָשִׁים הִשְׁפִּילוּ מַבָּט. וְדָוִד הִנְמִיךְ אֶת קוֹלוֹ עוֹד, לֹא כִּפְקֻדָּה אֶלָּא כֶּאֱמוּנָה: אִם אֶהְיֶה כְּמוֹתוֹ, בַּמֶּה אֶהְיֶה טוֹב מִמֶּנּוּ? לֹא בַּחֶרֶב אֶקַּח אֶת אֲשֶׁר נָתַן לִי ה' בַּשֶּׁמֶן.

וְאָז קָרָה הַדָּבָר, בְּלִי מִלָּה. דָּוִד עוֹדֶנּוּ עוֹמֵד, וְהַסַּכִּין שֶׁכָּרְתָה אֶת כְּנַף הַמְּעִיל בְּיָדוֹ הַיְמָנִית, מוּרֶמֶת וְרוֹטֶטֶת. לְאַט הִנְמִיךְ אוֹתָהּ. וּכְשֶׁהִגִּיעָה הַיָּד הַמּוּרֶמֶת אֶל מַטָּה, וְהִתְרַפְּתָה וְנִפְתְּחָה, נִשְׁמַט הַלַּהַב מִבֵּין אֶצְבְּעוֹתָיו אֶל הַקַּרְקַע בִּנְקִישַׁת מַתֶּכֶת יְחִידָה. וּבְאוֹתָהּ שְׁנִיָּה מַמָּשׁ, מִן הַצֵּל שֶׁמֵּאֲחוֹרָיו, הִנִּיחוּ יָדַיִם אֶל תּוֹךְ כַּף יָדוֹ הָרֵיקָה, הַפְּתוּחָה, אֶת הַכִּנּוֹר. אֵין הוֹרָאָה, אֵין דִּבּוּר, אֵין הֶסְבֵּר: יָד שֶׁאָחֲזָה בַּלַּהַב נִפְתְּחָה, וּמִתּוֹךְ אוֹתָהּ פְּתִיחָה מַמָּשׁ נוֹלַד הַכְּלִי. כְּאִלּוּ הַלַּהַב עַצְמוֹ נָמַס וְהָיָה לְכִנּוֹר. אֶצְבַּע אַחַת פָּרְטָה מֵיתָר בּוֹדֵד, וְהַצְּלִיל רָטַט בָּאֲוִיר. כָּל עֲצִירַת יָד מוֹלֶדֶת כִּנּוֹר.

הַכִּנּוֹר בְּיָדוֹ, וְהַמֵּיתָר הָאֶחָד עוֹדוֹ רוֹעֵד, כְּשֶׁשָּׁאוּל קָם בַּחוּץ, אָסַף אֶת מְעִילוֹ וַחֲנִיתוֹ, וְיָצָא מִן הַמְּעָרָה אֶל הָאוֹר — בְּלִי לָדַעַת דָּבָר. דָּוִד הִבִּיט אַחֲרָיו רֶגַע אָרֹךְ. רוּחוֹ סָעֲרָה בְּקִרְבּוֹ, וּכְמוֹ מֵאֵלָיו עָלָה בּוֹ הַתַּחֲנוּן: בְּקָרְאִי עֲנֵנִי אֱלֹהֵי צִדְקִי, בַּצָּר הִרְחַבְתָּ לִי. וְאָז, בְּטֶרֶם הִסְפִּיק לַחְשֹׁב, הִתְפָּרֵץ אֶל הַפֶּתַח.

אֲדֹנִי הַמֶּלֶךְ! קָרָא אֶל הַחוּץ.

שָׁאוּל קָפָא בְּאֶמְצַע צַעַד, וְהִסְתּוֹבֵב לְאַט. דָּוִד יָצָא אֶל הָאוֹר, הִשְׁתַּחֲוָה אַפַּיִם אַרְצָה, וְאָז הֵרִים גָּבוֹהַּ אֶת כְּנַף הַמְּעִיל, שֶׁתִּתְנוֹסֵס לְעֵינֵי שָׁאוּל. לָמָּה תִשְׁמַע אֶת דִּבְרֵי אָדָם לֵאמֹר, אָמַר, הִנֵּה דָוִד מְבַקֵּשׁ רָעָתֶךָ? אָבִי — רְאֵה! רְאֵה גַם רְאֵה אֶת כְּנַף מְעִילְךָ בְּיָדִי! כָּרַתִּי אֶת הַכָּנָף, וְלֹא הָרַגְתִּי אֶת הָאִישׁ. דַּע וּרְאֵה כִּי אֵין בְּיָדִי רָעָה וָפֶשַׁע, וְלֹא חָטָאתִי לָךְ — וְאַתָּה צֹדֶה אֶת נַפְשִׁי לְקַחְתָּהּ.

וְקוֹלוֹ עָלָה, לֹא בְּזַעַם כִּי אִם בִּכְאֵב שֶׁל בֵּן: אַחֲרֵי מִי יָצָא מֶלֶךְ יִשְׂרָאֵל? אַחֲרֵי מִי אַתָּה רֹדֵף — אַחֲרֵי כֶּלֶב מֵת? אַחֲרֵי פַּרְעֹשׁ אֶחָד? יִשְׁפֹּט ה' בֵּינִי וּבֵינֶךָ, וְיָדִי לֹא תִהְיֶה בָּךְ.

דְּמָמָה אֲרֻכָּה יָרְדָה. שָׁאוּל הִבִּיט בַּכָּנָף, בַּיָּד הַפְּרוּשָׂה, בִּפְנֵי דָוִד, וְהַקּוֹל הַגָּדוֹל, הַמַּלְכוּתִי, הִתְפּוֹרֵר. בְּקֹשִׁי, כִּמְעַט לֹא מַאֲמִין, שָׁאַל: הֲקֹלְךָ זֶה, בְּנִי דָוִד? וְאָז נִשְׁבַּר. נָשָׂא קוֹלוֹ וּבָכָה. צַדִּיק אַתָּה מִמֶּנִּי, אָמַר, כִּי אַתָּה גְּמַלְתַּנִי הַטּוֹבָה, וַאֲנִי גְּמַלְתִּיךָ הָרָעָה. שָׁתַק, וְאַחַר הוֹסִיף כִּמְעַט אֶל עַצְמוֹ: וְכִי יִמְצָא אִישׁ אֶת אֹיְבוֹ, וְשִׁלְּחוֹ בְּדֶרֶךְ טוֹבָה? וְכִי יֵשׁ אִישׁ כָּזֶה תַּחַת הַשָּׁמָיִם?

הוּא הִבִּיט בְּדָוִד אֲרֻכּוֹת, וּבָרֶגַע הַזֶּה הַיָּחִיד שֶׁל צְלִילוּת וְאַהֲבָה, פָּרְצָה הַנְּבוּאָה מִפִּיו כִּמְעַט בְּעַל כָּרְחוֹ: וְעַתָּה הִנֵּה יָדַעְתִּי כִּי מָלֹךְ תִּמְלוֹךְ, וְקָמָה בְּיָדְךָ מַמְלֶכֶת יִשְׂרָאֵל. צָעַד אֵלָיו, וְקוֹלוֹ נִשְׁבַּר: הִשָּׁבְעָה לִּי בַּה' אִם תַּכְרִית אֶת זַרְעִי אַחֲרַי, אִם תַּשְׁמִיד אֶת שְׁמִי מִבֵּית אָבִי.

נִשְׁבַּעְתִּי, הֵשִׁיב דָּוִד בְּשֶׁקֶט, יָדוֹ עַל לִבּוֹ.

רֶגַע עָמְדוּ שְׁנֵי אֲנָשִׁים זֶה מוּל זֶה — אָב שֶׁאִבֵּד אֶת דַּרְכּוֹ, וּבֵן שֶׁלֹּא הֵרִים יָד. וְאָז הִתְעַשֵּׁת שָׁאוּל, אָסַף אֶת שְׂרִידֵי כְּבוֹדוֹ, הֵסֵב פָּנָיו וְהָלַךְ. דָּוִד לֹא זָז עַד שֶׁנִּבְלַע צֵל הַמֶּלֶךְ בָּאוֹר. אֲנָשָׁיו יָצְאוּ בְּהַדְרָגָה, חֲרִישִׁיִּים, וְעָלוּ אֶל הַמְּצוּדָה, וְדָוִד נִשְׁאַר אַחֲרוֹן, לְבַדּוֹ, בְּמִפְתַּן הַמְּעָרָה.

שָׁאוּל הָלַךְ, וְהַשֶּׁקֶט שֶׁאַחֲרֵי הַחֶסֶד יָרַד. וְדַוְקָא עַכְשָׁו, כְּשֶׁאֵין עוֹד אוֹיֵב מוּל הָעֵינַיִם, חָזַר הַפַּחַד בְּכָל כֹּבְדּוֹ: דָּוִד עוֹדֶנּוּ נִרְדָּף, עוֹדֶנּוּ בַּמְּעָרוֹת, עוֹדֶנּוּ בְּלִי בָּיִת. הוּא נִשְׁעַן אֶל הַסֶּלַע הַקָּר, וּפִסַּת הַמְּעִיל עוֹד בְּיָדוֹ. וּמִתּוֹךְ הַשֵּׁפֶל הַזֶּה הֵחֵלָּה הָאֱמוּנָה לְטַפֵּס.

יָכֹלְתִּי, אָמַר אֶל הַחֹשֶׁךְ. הָיָה בְּיָדִי — וְהַיָּד לֹא רָעֲדָה מִפַּחַד. רָעֲדָה מִדָּבָר אַחֵר. הִבִּיט בַּכָּנָף. וְעַכְשָׁו? עַכְשָׁו הוּא הוֹלֵךְ לְבֵיתוֹ — וַאֲנִי? נָשָׂא עֵינָיו מַעְלָה. כַּמָּה זְמַן עוֹד, אֱלֹהַי? כַּמָּה לֵילוֹת בְּלִי שֵׁם, בֵּין הַסְּלָעִים, כְּשֶׁשִּׁנֵּי הָאֲנָשִׁים חֲנִית וְחִצִּים, וּלְשׁוֹנָם חֶרֶב חַדָּה?

וְכָאן פָּרַץ הַפָּסוּק, תְּחִלָּה כְּתַחֲנוּן מִן הַקַּרְקָעִית, וְקוֹלוֹ רַךְ:

> חָנֵּנִי אֱלֹהִים חָנֵּנִי — כִּי בְךָ חָסָיָה נַפְשִׁי,
> וּבְצֵל כְּנָפֶיךָ אֶחְסֶה — עַד יַעֲבֹר הַוּוֹת.
> אֶקְרָא לֵאלֹהִים עֶלְיוֹן — לָאֵל גֹּמֵר עָלָי.

הַקּוֹל הִתְחַזֵּק, וְהַתַּחֲנוּן הָפַךְ אַט אַט לְהַצְהָרָה. הַחֹשֶׁךְ נִשְׁאַר, אַךְ מַשֶּׁהוּ בּוֹ נִמְסַךְ בְּאוֹר:

> נָכוֹן לִבִּי אֱלֹהִים, נָכוֹן לִבִּי —
> אָשִׁירָה וַאֲזַמֵּרָה!
> עוּרָה כְבוֹדִי, עוּרָה הַנֵּבֶל וְכִנּוֹר —
> אָעִירָה שָּׁחַר!

וּבַשִּׂיא — לֹא בַּקָּשַׁת הַצָּלָה, אֶלָּא שֶׁבַח שֶׁמִּתְפָּרֵץ מֵעֵבֶר לַפַּחַד:

> כִּי גָדֹל עַד שָׁמַיִם חַסְדֶּךָ —
> וְעַד שְׁחָקִים אֲמִתֶּךָ.
> רוּמָה עַל שָׁמַיִם אֱלֹהִים —
> עַל כָּל הָאָרֶץ כְּבוֹדֶךָ!

וְהָאוֹר עָלָה אַט אַט עִם הַשִּׁיר — לֹא אוֹר שֶׁל יְשׁוּעָה חִיצוֹנִית, אֶלָּא אוֹר פְּנִימִי שֶׁדָּוִד נָשָׂא אֶל תּוֹךְ הַחֹשֶׁךְ. הוּא הִדֵּק אֶת פִּסַּת הַמְּעִיל בְּיָדוֹ, הִסְתּוֹבֵב, וְעָלָה אֶל אֲנָשָׁיו, וְהַכִּנּוֹר עִמּוֹ.
