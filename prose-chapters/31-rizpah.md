# פֶּרֶק ל: רִצְפָּה בַּת אַיָּה

עַל הַצּוּק הֻנְּחוּ שִׁבְעָה גּוּפוֹת — בָּנֶיהָ וּבְנֵי שָׁאוּל — וְהָאָבִיב פָּרַח סְבִיבָם בְּלֹא בּוּשָׁה. וְאַחַר בָּאוּ הַגְּשָׁמִים, מַלְקוֹשׁ, וְעוֹד מַלְקוֹשׁ, חֳדָשִׁים אֲרֻכִּים. וְרִצְפָּה בַּת אַיָּה לְבַדָּהּ עַל הַצּוּק, וְשַׂק פָּרוּשׂ לָהּ עַל הַסֶּלַע, יוֹם אַחַר יוֹם.

בָּאוּ הָעוֹפוֹת — וְהִיא גֵּרְשָׁה אוֹתָם. בָּאוּ חַיּוֹת הַלַּיְלָה — וְהִיא גֵּרְשָׁה אוֹתָן. גֶּשֶׁם. שֶׁמֶשׁ. וְגֶשֶׁם שׁוּב. וְהִיא שָׁם. לֹא נָאֲמָה, לֹא קוֹנְנָה בְּקוֹל. רַק נָכְחָה. וְזֶה הָיָה כֹּחָהּ.

*

בִּירוּשָׁלַיִם נִכְנַס אֶחָד מֵאַנְשֵׁי דָּוִד אֶל הַמֶּלֶךְ. "אֲדֹנִי," אָמַר, "שָׁמַעְנוּ דְּבַר רִצְפָּה בַּת אַיָּה..." וְסִפֵּר. וְדָוִד שָׁמַע, וְשָׁתַק. וְאָז קָם, וְאָמַר בְּשֶׁקֶט:

"קִבְרוּ אֶת עַצְמוֹת שָׁאוּל וִיהוֹנָתָן בְּנוֹ בְּקֶבֶר קִישׁ אֲבִי שָׁאוּל... וְאֵת עַצְמוֹת הַמּוּקָעִים. וַעֲשׂוּ אֵת כָּל אֲשֶׁר צִוָּה הַמֶּלֶךְ." (שמ"ב כא, יג-יד)

*

עַל הַצּוּק רָאֲתָה רִצְפָּה אֲנָשִׁים מַגִּיעִים. לֹא זָזָה — עַד שֶׁרָאֲתָה כִּי אֵינָם אוֹיְבִים. נִגַּשׁ דָּוִד אֵלֶיהָ, וְלֹא דִּבֵּר. וְהִיא לֹא דִּבְּרָה. מֶלֶךְ מוּל אֵם שַׁכּוּלָה — וְאֵין מִלָּה שֶׁתַּחֲזִיק אֶת שְׁנֵיהֶם.

הִבִּיט בַּגּוּפוֹת, וּבַשַּׂק הַפָּרוּשׂ עַל הַסֶּלַע, וּבְיָדֶיהָ — שָׁחֹר תַּחַת הַצִּפָּרְנַיִם, מִן הַגֵּרוּשׁ שֶׁל חֳדָשִׁים. וְהוּא, אֲשֶׁר הָרַג וְהֻמְלַךְ וּבָרַח וְשָׁב, הֵבִין לְרֶגַע כִּי אִשָּׁה אַחַת, שֶׁאֵין בְּיָדָהּ כֹּחַ לְשַׁנּוֹת גְּזֵרָה, נִצְּחָה אוֹתָהּ בְּכָךְ שֶׁלֹּא עָזְבָה.

הִבִּיט בְּיָדֶיהָ הַשְּׁחֹרוֹת — וְאָז בְּיָדָיו שֶׁלּוֹ. כָּל יָמָיו עָצַר אֶת יָדוֹ מֵהַכּוֹת: בַּמְּעָרָה, אֵצֶל נָבָל, אֵצֶל שִׁמְעִי. וּמִתּוֹךְ הָעֲצִירָה נוֹלַד כִּנּוֹר. וְהִיא לֹא הִרְפְּתָה אֶת יָדֶיהָ מִן הַשְּׁמִירָה — וּמִתּוֹךְ אִי הַהַרְפָּיָה נוֹלַד חֶסֶד. שְׁתֵּי יָדַיִם שֶׁאֵינָן עוֹשׂוֹת אֶת הַמַּעֲשֶׂה הַמִּתְבַּקֵּשׁ, לְהַכּוֹת אוֹ לַעֲזֹב, וּשְׁתֵּיהֶן מוֹלִידוֹת. וְאָז סִיֵּם אֶת אֲשֶׁר הִיא הֵחֵלָּה: קָבַר.

וְכִנּוֹר עָלָה — וְאִם דִּבֵּר דָּוִד, דִּבֵּר בְּשֶׁקֶט, אֵלֶיהָ וְלֹא אֵלֶיהָ:

> לֹא בְּחֶרֶב שָׁמַרְתְּ — בְּגוּפֵךְ.
> לֹא בְּצָבָא — בִּשְׁתֵּי יָדַיִם.
> כָּל הַמְּלָכִים מִתְהַלְּלִים בְּמַה שֶּׁכָּבְשׁוּ —
> וְאַתְּ — בְּמַה שֶּׁסֵּרַבְתְּ לְהַנִּיחַ.
> אֲנִי בָּנִיתִי מַמְלָכָה.
> אַתְּ שָׁמַרְתְּ עַל שֶׁבַע גְּוִיּוֹת מֵהָעוֹרֵב.
> וְאֵינֶנִּי יוֹדֵעַ מִי מִשְּׁנֵינוּ גָּדוֹל.

קָבְרוּ. רִצְפָּה עָמְדָה. וְאָז הָלְכָה.

*

אַחֲרֵי שֶׁהָלְכָה נוֹתַר דָּוִד עִם אֲנָשָׁיו, וּבָא אֶל הַקְּרָב נֶגֶד יְלִידֵי הָרָפָה. וַתְּהִי עוֹד מִלְחָמָה לַפְּלִשְׁתִּים אֶת יִשְׂרָאֵל, וַיֵּרֶד דָּוִד וַעֲבָדָיו עִמּוֹ — וַיָּעַף דָּוִד. (שמ"ב כא, טו)

יָשַׁב בְּצַד הַקְּרָב. לֹא נָפַל — אַךְ גַּם לֹא קָם. רָאוּ עָלָיו אֶת הָעֲיֵפוּת, גּוּפָנִית, וְזֶה הָיָה דָּבָר חָדָשׁ. חַיִל פְּלִשְׁתִּי הִתְנַפֵּל עָלָיו, וַאֲבִישַׁי חָסַם אֶת פָּנָיו וְהִצִּילוֹ.

וְאַחֲרֵי הַקְּרָב נִגְּשׁוּ הַשָּׂרִים אֶל דָּוִד: "לֹא תֵצֵא עוֹד אִתָּנוּ לַמִּלְחָמָה, וְלֹא תְכַבֶּה אֶת נֵר יִשְׂרָאֵל." (שמ"ב כא, יז)

הִבִּיט עֲלֵיהֶם דָּוִד וְלֹא הִתְוַכֵּחַ. יָדַע שֶׁצָּדְקוּ.
