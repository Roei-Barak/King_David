# גִּלְבֹּעַ — נְפִילַת הַמֶּלֶךְ

הָרוּחַ לֹא נָשְׁבָה בְּהַר הַגִּלְבֹּעַ בְּאוֹתוֹ יוֹם. הַשָּׁמַיִם הָיוּ לְבָנִים, קָרִים, שְׁתוּקִים — כְּאִלּוּ גַּם הַשָּׁמַיִם הֵסִיבוּ פְּנֵיהֶם. וְהַחִצִּים עָלוּ וְיָרְדוּ כִּגְשֶׁם שֶׁאֵין בּוֹ רַחֲמִים.

אַנְשֵׁי יִשְׂרָאֵל נָסוּ.

לֹא כְּגִבּוֹרִים הַנּוֹסוֹגִים לִנְשֹׁם, כִּי אִם כְּמִי שֶׁנִּשְׁבַּר בָּהֶם דָּבָר שֶׁלֹּא יַחֲזֹר. הֵם נָפְלוּ עַל מַדְרוֹן הָהָר, אֶחָד אֶחָד, עַד שֶׁהָרוּחַ פָּסְקָה וְנִשְׁאַר רַק הַקּוֹל הַכָּבֵד שֶׁל פְּלִשְׁתִּים מְנַצְּחִים וְהַר שֶׁשָּׁתַק.

וְשָׁאוּל — הָיָה עוֹמֵד.

עוֹד עָמַד. גּוּפוֹ הַגָּבוֹהַּ, שֶׁמִּכְּתֵפָיו וָמַעְלָה גָּבַהּ מִכָּל הָעָם, עָמַד בִּזְקִיפוּת שֶׁל אִישׁ שֶׁאֵינוֹ יוֹדֵעַ עוֹד אֵיךְ כּוֹרְעִים. אַךְ הַחֵץ כְּבָר מָצָא אוֹתוֹ. הוּא הִרְגִּישׁ אֶת הַמַּכָּה — חַמָּה וּמְשֻׁנָּה, כִּמְחָט שֶׁמְּפָרֶרֶת אֶת הַגּוּף מִבִּפְנִים — וְיָדַע.

*יְהוֹנָתָן.*

הַמִּלָּה עָלְתָה בְּלֹא שֶׁנָּעוּ שְׂפָתָיו. בְּנוֹ. בְּנוֹ הַגָּדוֹל. אוֹתוֹ אָהַב יוֹתֵר מִן הַמַּלְכוּת, יוֹתֵר מִן הַחֶרֶב, כִּמְעַט יוֹתֵר מִשֶּׁשָּׂנֵא אֶת בֶּן יִשַׁי. וְהִנֵּה — בְּיוֹם אֶחָד. לֹא נִפְרְדוּ בְּחַיֵּיהֶם. לֹא יִפָּרְדוּ בְּמוֹתָם.

אָמַר יְהוֹנָתָן פַּעַם: *אַתָּה תִּמְלֹךְ.* וְהוּא, שָׁאוּל, אָמַר: *לֹא כָּל עוֹד הַנַּעַר חַי.* וַיִּרְאֵהוּ עַתָּה, בְּרֶגַע הָאַחֲרוֹן, כָּל הַשָּׁנִים אַחַת אַחַת — הַנַּעַר שֶׁלּוֹ, שֶׁרָץ אֵלָיו בְּשֶׂמֶץ כֵּן שֶׁאֵין לָדַעַת מֵהֵיכָן בָּא, שֶׁאָהַב אֶת אוֹיְבוֹ כְּנַפְשׁוֹ. וְלֹא יְשַׁלֵּם לוֹ עָלָיו כְּלוּם.

"כָּל יָמַי בַּקֶּשֶׁת", אָמַר שָׁאוּל בְּקוֹל שֶׁנִּשְׁאַר בְּתוֹכוֹ. "מִיּוֹם מָשְׁחֵנִי שְׁמוּאֵל וְעַד הַיּוֹם. וְהַיּוֹם הַקֶּשֶׁת מָצְאָה אוֹתִי."

חָשַׁב — בְּכָבוֹד יִפֹּל. כְּמֶלֶךְ. לֹא כָּכָה.

נוֹשֵׂא כֵּלָיו עָמַד סָמוּךְ, פָּנָיו חִוְּרִים כְּסִיד, וְשָׁאוּל פָּנָה אֵלָיו בְּעֵינַיִם שֶׁרָאוּ עוֹד.

"שְׁלֹף חַרְבְּךָ וְדָקְרֵנִי בָהּ," אָמַר, "פֶּן יָבֹאוּ הָעֲרֵלִים הָאֵלֶּה וּדְקָרוּנִי וְהִתְעַלְּלוּ בִי."

הַנַּעַר לֹא זָז. יָדָיו רָעֲדוּ.

"אֲדֹנִי הַמֶּלֶךְ... לֹא אוּכַל. יָדִי לֹא תִּגַּע בְּךָ — כִּי מְשִׁיחַ ה' אָתָּה."

שָׁאוּל הִבִּיט בּוֹ רֶגַע. לֹא בְּכַעַס. בְּמַשֶּׁהוּ שֶׁהָיָה יְכוֹל לִהְיוֹת רַחֲמִים, אִלּוּ הִיָּה עוֹד זְמַן לְרַחֲמִים.

לָקַח אֶת הַחֶרֶב.

וְהַמִּלִּים שֶׁנָּשַׁר לְאוֹתָן שְׁנִיּוֹת — לֹא נֶאֶמְרוּ לְאִישׁ. הָיוּ כְּשִׁירָה שֶׁנּוֹשֵׁרֶת מֵאִישׁ שֶׁלֹּא לָמַד לָשִׁיר:

לְבַד.\
תָּמִיד הָיִיתִי לְבַד —\
מִיּוֹם שֶׁבָּרְחָה רוּחַ ה' מִמֶּנִּי.

אֱלֹהִים —\
לֹא עָנִיתָ לִי.\
לֹא בַּחֲלוֹם,\
לֹא בְּאוּרִים,\
לֹא בִּנְבִיאִים.\
שָׁתַקְתָּ לִי כָּל חַיַּי הָאַחֲרוֹנִים.

וְעַתָּה —\
בַּשְּׁתִיקָה הַזֹּאת אֶפֹּל.\
לֹא בְּחֶרֶב אוֹיֵב —\
בְּחַרְבִּי שֶׁלִּי.

אַל תִּירָא, נַעַר.\
לֹא חָטָאתָ.\
טוֹב עָשִׂיתָ שֶׁלֹּא נָגַעְתָּ בִי.

בֶּן יִשַׁי —\
הַמַּמְלָכָה שֶׁלְּךָ הִיא.\
תֵּן לָהּ לֵב טוֹב\
מִשֶּׁלִּי.

נָפַל שָׁאוּל עַל חַרְבּוֹ.

וְנָפַל גַּם נוֹשֵׂא הַכֵּלִים אַחֲרָיו — כִּי לֹא יָכֹל לַעֲמֹד בְּעוֹלָם שֶׁאֵין בּוֹ מֶלֶךְ.

הָהָר הִשְׁתַּתֵּק. הַשָּׁמַיִם הַלְּבָנִים הִשְׁתַּתְּקוּ. וְהַפְּלִשְׁתִּים מָצְאוּ אוֹתָם בְּמָחֳרָת, שְׁלָשְׁתָּם, שָׁאוּל וּשְׁלֹשֶׁת בָּנָיו, עַל שִׁפּוּלֵי הַגִּלְבֹּעַ. וַיִּקְחוּ אֶת רֹאשׁוֹ וַיִּתְלוּ אֶת גּוּפוֹתֵיהֶם עַל חוֹמוֹת בֵּית שְׁאָן.

---

אַךְ בְּלֵיל אֶחָד — לַיְלָה אֶחָד בִּלְבַד — קָמוּ אַנְשֵׁי יָבֵשׁ גִּלְעָד.

זָכְרוּ. זָכְרוּ אֶת הַיּוֹם שֶׁהָיוּ בְּצָרָה, עִיר נְצוּרָה, וְהָאִישׁ שֶׁלֹּא הָיָה עוֹד צָעִיר וְלֹא עוֹד בּוֹטֵחַ בְּעַצְמוֹ — קָם. וְהוֹשִׁיעַ. לֹא בִּשְׁבִיל הַתְּהִלָּה. בִּשְׁבִיל הָאֲנָשִׁים.

אִישׁ אֶחָד, שֵׂיבָה בִּזְקָנוֹ, אָמַר לַחֲבֵרָיו בְּלַחַשׁ: "לֹא נַנִּיחַ אוֹתוֹ עַל הַחוֹמָה לְבֹשֶׁת."

וְלֹא אָמְרוּ דָּבָר אַחֵר. לֹא הָיָה צֹרֶךְ.

הֵם הָלְכוּ כָּל הַלַּיְלָה.

בְּחֹשֶׁךְ, בְּדֹמִיָּה, בְּדֶרֶךְ שֶׁהֵם לְבַדָּם יָדְעוּ — חָצוּ אֶת הָעֵמֶק, עָלוּ אֶל הַחוֹמָה, וְנָשְׂאוּ עַל כִּתְפֵיהֶם אֵת שֶׁנִּשְׁאַר מִן הַמֶּלֶךְ. מִן הַיָּלִיד. מִן הָאִישׁ שֶׁלֹּא יָדַע לִמְחֹל אַךְ יָדַע לְהוֹשִׁיעַ.

קָבְרוּ אוֹתָם תַּחַת הָאֵשֶׁל בְּיָבֵשָׁה. וְצָמוּ שִׁבְעַת יָמִים — כִּי כָּךְ יֵשׁ לָעֲשׂוֹת לַאֲנָשִׁים שֶׁהָיוּ גְּדוֹלִים, אֲפִלּוּ כְּשֶׁנָּפְלוּ.

בְּאֵיזֶה מָקוֹם, רָחוֹק מִשָּׁם, בֵּין הַצִּקְלַג וּבֵין הֶעָתִיד שֶׁעוֹד לֹא יָדַע שֶׁהוּא שֶׁלּוֹ — דָּוִד עוֹד לֹא יָדַע.

עֲדַיִן לֹא בָּא אֵלָיו הַמְּבַשֵּׂר. עֲדַיִן לֹא יָדַע שֶׁנִּגְמַר הַמַּתָּנָה. עֲדַיִן לֹא יָדַע שֶׁהוּא כְּבָר מֶלֶךְ.
