# פֶּרֶק יא: הַפְּלִשְׁתִּים דּוֹחִים

בֹּקֶר עָלָה עַל שְׂדֵה הַכִּנּוּס הַפְּלִשְׁתִּי בַּאֲפֵק. סָרְנֵי פְלִשְׁתִּים עָבְרוּ לְסִדְרֵיהֶם — לְמֵאוֹת וְלַאֲלָפִים — וּבָעֹרֶף, בְּקָצֶה אַחֵר, צָעֲדוּ דָּוִד וַאֲנָשָׁיו עִם אַכִישׁ. זֶה הָיָה הַמַּלְכֹּד הַנּוֹרָא שֶׁל חַיֵּי דָּוִד בַּגָּלוּת: הוּא בָּרַח אֶל אַכִישׁ, קִבֵּל אֶת צִקְלַג, וּכְדֵי לִשְׂרֹד הִצִּיג עַצְמוֹ כְּעֶבֶד נֶאֱמָן לַפְּלִשְׁתִּים. עַתָּה נִפְרַעַת הַתַּחְבּוּלָה, וְדָוִד עוֹמֵד לִצְעֹד עִם הָאוֹיֵב נֶגֶד עַמּוֹ, נֶגֶד שָׁאוּל מְשִׁיחַ ה', נֶגֶד יְהוֹנָתָן אֲהוּבוֹ.

מַבְּטֵי הַסְּרָנִים נָפְלוּ עֲלֵיהֶם, וְרַחַשׁ חֲשָׁד עָלָה. אֶחָד מֵהֶם נֶעֱצַר, הִצְבִּיעַ עַל דָּוִד מֵרָחוֹק, וְקוֹלוֹ הִתְחַדֵּד: "מָה הָעִבְרִים הָאֵלֶּה?! (שמ"א כט, ג) מַה לְעִבְרִי בְּמַחֲנֵה פְלִשְׁתִּים?!"

אַכִישׁ הִתְיַצֵּב, יָדוֹ עַל כֶּתֶף דָּוִד — תָּמִים, גֵּאֶה בְּעַבְדּוֹ. "הֲלוֹא זֶה דָוִד עֶבֶד שָׁאוּל מֶלֶךְ יִשְׂרָאֵל, אֲשֶׁר הָיָה אִתִּי זֶה יָמִים אוֹ זֶה שָׁנִים, וְלֹא מָצָאתִי בוֹ מְאוּמָה מִיּוֹם נָפְלוֹ עַד הַיּוֹם הַזֶּה!" (כט, ג)

אַךְ הַסְּרָנִים לֹא נָחוּ. פְּנֵיהֶם קָדְרוּ, וְאֶחָד פָּסַע קָדִימָה. קוֹלוֹ רָעַם בְּכַעַס אֶל אַכִישׁ, וְהִתְעַלֵּם מִדָּוִד כְּאִלּוּ אֵינֶנּוּ. "הָשֵׁב אֶת הָאִישׁ! וְיָשֹׁב אֶל מְקוֹמוֹ אֲשֶׁר הִפְקַדְתּוֹ שָׁם, וְלֹא יֵרֵד עִמָּנוּ בַּמִּלְחָמָה — וְלֹא יִהְיֶה לָנוּ לְשָׂטָן בַּמִּלְחָמָה! (כט, ד) וּבַמֶּה יִתְרַצֶּה זֶה אֶל אֲדֹנָיו — הֲלוֹא בְּרָאשֵׁי הָאֲנָשִׁים הָהֵם?!" (כט, ד)

וְהָרִאשׁוֹן הוֹסִיף, מֵסִית, מַצְבִּיעַ בְּאֶצְבַּע מְאַשֶּׁמֶת: "הֲלוֹא זֶה דָוִד — אֲשֶׁר יַעֲנוּ לוֹ בַּמְּחֹלוֹת לֵאמֹר: 'הִכָּה שָׁאוּל בַּאֲלָפָיו — וְדָוִד בְּרִבְבֹתָיו!'" (כט, ה)

שְׁתִיקָה. הַשֵּׁם "דָּוִד" צָף בָּאֲוִיר כְּאִיּוּם. הַסְּרָנִים הָיוּ מְאֻחָדִים בַּחֲשָׁדָם, וְאַכִישׁ, מֻבָּס, הֵבִין שֶׁאֵין לוֹ בְּרֵרָה. דָּוִד עָמַד מִן הַצַּד, פָּנָיו אֶבֶן — אַךְ עֵינָיו, רַק לְרֶגַע, נָשְׁמוּ.

*

אַכִישׁ קָרָא לְדָוִד אֵלָיו, מִצְטַעֵר, כִּמְעַט מִתְנַצֵּל, וְיָדָיו פְּרוּשׂוֹת בְּכֵנוּת פְּלִשְׁתִּית תְּמִימָה. "חַי ה' כִּי יָשָׁר אַתָּה, וְטוֹב בְּעֵינַי צֵאתְךָ וּבֹאֲךָ אִתִּי בַּמַּחֲנֶה, כִּי לֹא מָצָאתִי בְךָ רָעָה מִיּוֹם בֹּאֲךָ אֵלַי עַד הַיּוֹם הַזֶּה — וּבְעֵינֵי הַסְּרָנִים לֹא טוֹב אָתָּה. (כט, ו) וְעַתָּה שׁוּב וְלֵךְ בְּשָׁלוֹם, וְלֹא תַעֲשֶׂה רָע בְּעֵינֵי סַרְנֵי פְלִשְׁתִּים." (כט, ז)

וְדָוִד, נִפְגָּע כְּלַפֵּי חוּץ, לָבַשׁ אֶת עֶלְבּוֹן הָעֶבֶד הַנֶּאֱמָן. מִשְׂחָק מֻפְלָא הָיָה זֶה: הוּא יָדַע שֶׁנִּצַּל, אַךְ חַיָּב הָיָה לְהַפְגִּין אַכְזָבָה, וְקוֹלוֹ פָּצוּעַ. "כִּי מֶה עָשִׂיתִי? וּמַה מָּצָאתָ בְעַבְדְּךָ מִיּוֹם אֲשֶׁר הָיִיתִי לְפָנֶיךָ עַד הַיּוֹם הַזֶּה — כִּי לֹא אָבוֹא וְנִלְחַמְתִּי בְּאֹיְבֵי אֲדֹנִי הַמֶּלֶךְ?!" (כט, ח)

הוּא הִשְׁפִּיל מַבָּט, כִּמְאֻכְזָב מָרָה, וְאַכִישׁ הִנִּיחַ יָד מְנַחֶמֶת עַל שִׁכְמוֹ, רַךְ וּמְבַקֵּשׁ סְלִיחָה. "יָדַעְתִּי כִּי טוֹב אַתָּה בְּעֵינַי כְּמַלְאַךְ אֱלֹהִים — אַךְ שָׂרֵי פְלִשְׁתִּים אָמְרוּ: לֹא יַעֲלֶה עִמָּנוּ בַּמִּלְחָמָה. (כט, ט) וְעַתָּה הַשְׁכֵּם בַּבֹּקֶר... וְאוֹר לָכֶם וָלֵכוּ." (כט, י)

אַכִישׁ פָּנָה לָלֶכֶת. דָּוִד נִשְׁאַר, רֹאשׁוֹ עוֹד שָׁמוּט בְּעֶלְבּוֹן מֻעֲמָד — עַד שֶׁאַכִישׁ נֶעֱלַם.

*

וְכָאן, לְבַדּוֹ, נָשַׁר הָעֶלְבּוֹן מֵעַל פָּנָיו כְּמַסֵּכָה. בְּרֶגַע שֶׁאִישׁ פְּלִשְׁתִּי אֵינוֹ רוֹאֶה אוֹתוֹ, נָשַׁם דָּוִד נְשִׁימָה אֲרֻכָּה, כְּמוֹ נִצָּל מִטְּבִיעָה. פָּנָה כְּלַפֵּי מַעְלָה, וְהַמִּלִּים עָלוּ בּוֹ כַּהֲקָלָה גְּדוֹלָה, סוֹד שֶׁאָסוּר לַחֲלֹק עִם אִישׁ. תְּחִלָּה לָחַשׁ, וְאָז גָּדַל הַקּוֹל:

> בָּרוּךְ אַתָּה...
> אֲשֶׁר שַׂמְתָּ שִׂנְאַת אוֹיְבַי לִי לְמָגֵן,
> וּמִפִּי הַחֲשֵׁדִים הוֹצֵאתָ לִי יְשׁוּעָה.
>
> הֵם דְּחָאוּנִי — וְלֹא יָדְעוּ כִּי הִצַּלְתָּנִי.
> חָשְׁבוּ עָלַי רָעָה — וְהָיְתָה לִי לְפֶתַח.
> כִּי לוּלֵא הֵם — בַּמֶּה הָיִיתִי בָא הַבֹּקֶר?
> הַאֶשְׁלַח יָד בְּמַחֲנֵה עַמִּי?
> הַאֶצְעַד עַל שָׁאוּל מְשִׁיחֲךָ, עַל יְהוֹנָתָן אָחִי?

עֵינָיו נֶעֶצְמוּ.

> חָלִילָה לִּי — וְלֹא יָדַעְתִּי דֶּרֶךְ לָצֵאת.
> וְאַתָּה — מִתּוֹךְ פִּי שׂוֹנְאַי פָּתַחְתָּ לִי שַׁעַר.

וּמִתּוֹךְ עֹמֶק הַהֲקָלָה עָלוּ אֶל פָּנָיו דִּבְרֵי הַתְּהִלִּים, כְּמַעְיָן שֶׁנִּפְתַּח:

> אַל תֵּרַע לְךָ בַּמְּרֵעִים... כִּי כֶחָצִיר מְהֵרָה יִמָּלוּ. (תהילים לז, א-ב)
> בְּטַח בַּה' וַעֲשֵׂה טוֹב — שְׁכָן אֶרֶץ וּרְעֵה אֱמוּנָה. (לז, ג)
> גּוֹל עַל ה' דַּרְכֶּךָ, וּבְטַח עָלָיו — וְהוּא יַעֲשֶׂה. (לז, ה)

חִיּוּךְ דַּק, מָהוּל בְּדֶמַע, עָלָה עַל שְׂפָתָיו:

> הֵם חָשְׁבוּ לְגָרְשֵׁנִי —
> וְאַתָּה גֵּרַשְׁתַּנִי אֶל הַחַיִּים.

הָאוֹר נִשְׁאַר עָלָיו לְבַדּוֹ רֶגַע אָרֹךְ. וְאָז — חֹשֶׁךְ.
